\state{(Jackson 14.3)}{
	The Heaviside-Feynman expression for the electric field of a particle of charge $e$ in arbitrary motion, an alternative to the {\Lienard}-Wiechert expression in Eq.~\refeq{LWfields}, is
	\eqn{HF}{
		\vE = e \brac{ \frac{\nh}{R^2} }\ret + e \brac{ \frac{R}{c} }\ret \dv{t} \brac{ \frac{\nh}{R^2} }\ret + \frac{e}{c^2} \dv[2]{t} [ \nh ]\ret,
	}
	where the time derivatives are with respect to the time at the observation point.  Using the fact that the retarded time is $t' = t - R(t') / c$ and that, as a result,
	\eq{
		\dv{t}{t'} = 1 - \vbet(t') \vdot \nh(t'),
	}
	show that the form above yields the expression for $\vE$ in Eq.~\refeq{LWfields} when the time differentiations are performed.
}

\sol{
	One of the vector identities on the inside cover of Jackson is
	\eq{
		\vaa \cross (\vbb \cross \vcc) = (\vaa \vdot \vcc) \vbb - (\vaa \vdot \vbb) \vcc.
	}
	From this,
	\al{
		\nh \cross [ (\nh - \vbet) \cross \vbetd ] &= (\nh \vdot \vbetd) (\nh - \vbet) - [\nh \vdot (\nh - \vbet) ] \vbetd
		= (\nh \vdot \vbetd) \nh - (\nh \vdot \vbetd) \vbet - \vbetd + (\nh \vdot \vbet) \vbetd \\
		&= \frac{(\nh \vdot \vvd) \nh}{c} - \frac{(\nh \vdot \vvd) \vv}{c^2} - \frac{\vvd}{c} + \frac{(\nh \vdot \vv) \vvd}{c^2}.
	}
	Then Eq.~\refeq{LWfields} can be written
	\al{
		\vE &= e \brac{ \frac{1}{\gam^2 (1 - \vbet \vdot \nh)^3 R^2} \paren{ \nh - \frac{\vv}{c} } }\ret + \frac{e}{c} \brac{ \frac{1}{(1 - \vbet \vdot \nh)^3 R} \paren{ \frac{(\nh \vdot \vvd) \nh}{c} - \frac{(\nh \vdot \vvd) \vv}{c^2} - \frac{\vvd}{c} + \frac{(\nh \vdot \vv) \vvd}{c^2} } }\ret \\
		&= e \brac{ \frac{1}{(1 - \vbet \vdot \nh)^3 R^2} \paren{ \frac{\nh}{\gam^2} - \frac{\vv}{c \gam^2} + \frac{R (\nh \vdot \vvd) \nh}{c^2} - \frac{R (\nh \vdot \vvd) \vv}{c^3} - \frac{R \vvd}{c^2} + \frac{R (\nh \vdot \vv) \vvd}{c^3} } }\ret \\
		&= \frac{e}{c^3} \brac{ \frac{1}{(1 - \vbet \vdot \nh)^3 R^2} \paren{ \frac{c^3 \nh}{\gam^2} - \frac{c^2 \vv}{\gam^2} + c R (\nh \vdot \vvd) \nh - R (\nh \vdot \vvd) \vv - c R \vvd + R (\nh \vdot \vv) \vvd } }\ret.
	}

	Since $R(t') = c(t - t')$, note that
	\eq{
		\brac{ \dv{R}{t} }\ret = \dv{R(t')}{t'}
		= c \paren{ \dv{t}{t'} - 1 }
		= c (1 - \vbet(t') \vdot \nh(t') - 1)
		= -[ \nh \vdot \vv ]\ret.
	}
	Note that $[ \nh ]\ret = [ (\vx - \vr) / R ]\ret$.  Then
	\al{
		\dv{t'}[ \nh ]\ret &= \brac{ \dv{\nh}{t} }\ret
		= \brac{ \frac{1}{R^2} \paren{ R \dv{t}(\vx - \vr) - (\vx - \vr) \dv{R}{t} } }\ret
		= \brac{ \frac{1}{R^2} \paren{ -R \vv + R (\nh \vdot \vv) \nh } }\ret \\
		&= \brac{ \frac{(\nh \vdot \vv) \nh - \vv}{R} }\ret, \\[2ex]
		%
		\dv{t'}\!\brac{ \frac{\nh}{R} }\ret &= \brac{ \dv{t}(\frac{\nh}{R}) }\ret
		= \brac{ \frac{1}{R^2} \paren{ R \dv{\nh}{t} - \nh \dv{R}{t} } }\ret
		= \brac{ \frac{1}{R^2} \curly{ R \paren{ \frac{(\nh \vdot \vv) \nh - \vv}{R} } + (\nh \vdot \vv) \nh } }\ret \\
		&= \brac{ \frac{(\nh \vdot \vv) \nh - \vv + (\nh \vdot \vv) \nh}{R^2} }\ret
		= \brac{ \frac{2 (\nh \vdot \vv) \nh - \vv}{R^2} }\ret, \\[2ex]
		%
		\dv{t'}\!\brac{ \frac{\nh}{R^2} }\ret &= \brac{ \dv{t}(\frac{\nh}{R^2}) }\ret
		= \brac{ \frac{1}{R^2} \curly{ R \dv{t}(\frac{\nh}{R}) - \frac{\nh}{R} \dv{R}{t} } }\ret
		= \brac{ \frac{1}{R^2} \paren{ R \frac{2 (\nh \vdot \vv) \nh - \vv}{R^2} + \frac{(\nh \vdot \vv) \nh}{R} } }\ret \\
		&= \brac{ \frac{2 (\nh \vdot \vv) \nh - \vv + (\nh \vdot \vv) \nh}{R^3} }\ret
		= \brac{ \frac{3 (\nh \vdot \vv) \nh - \vv}{R^3} }\ret.
	}
	
	For the second derivative of $\nh$, note that
	\eq{
		\dv{t'}[ \nh \vdot \vv ]\ret = \brac{ \dv{t}(\nh \vdot \vv) }\ret
		= \brac{ \nh \vdot \dv{\vv}{t} + \dv{\nh}{t} \vdot \vv }\ret
		= \brac{ \nh \vdot \vvd + \frac{(\nh \vdot \vv) \nh - \vv}{R} \vdot \vv }\ret
		= \brac{ \nh \vdot \vvd + \frac{(\nh \vdot \vv)^2 - \vv^2}{R} }\ret,% \\[2ex]
	}
	so then
	\al{
		\dv[2]{{t'}}[ \nh ]\ret &= \brac{ \dv[2]{\nh}{t} }\ret
		= \brac{ \dv{t}(\frac{(\nh \vdot \vv) \nh - \vv}{R}) }\ret
		= \brac{ \dv{t}(\frac{(\nh \vdot \vv) \nh}{R}) - \dv{t}(\frac{\vv}{R}) }\ret \\
		&= \brac{ \frac{\nh}{R} \dv{t}(\nh \vdot \vv) + (\nh \vdot \vv) \dv{t}(\frac{\nh}{R}) - \frac{1}{R^2} \paren{ R \dv{\vv}{t} - \vv \dv{R}{t} } }\ret \\
		&= \brac{ \frac{\nh}{R} \paren{ \nh \vdot \vvd + \frac{(\nh \vdot \vv)^2 - \vv^2}{R} } + (\nh \vdot \vv) \paren{ \frac{2 (\nh \vdot \vv) \nh - \vv}{R^2} } - \frac{1}{R^2} \paren{ R \vvd + (\nh \vdot \vv) \vv } }\ret \\
		&= \brac{ \frac{(\nh \vdot \vv) \nh}{R} + \frac{(\nh \vdot \vv)^2 \nh}{R^2} - \frac{\vv^2 \nh}{R^2} + 2 \frac{(\nh \vdot \vv)^2 \nh}{R^2} - \frac{(\nh \vdot \vv) \vv}{R^2} - \frac{\vvd}{R} - \frac{(\nh \vdot \vv) \vv}{R^2} }\ret \\
		&= \brac{ \frac{(\nh \vdot \vv) \nh}{R} + 3 \frac{(\nh \vdot \vv)^2 \nh}{R^2} - \frac{\vv^2 \nh}{R^2} - 2 \frac{(\nh \vdot \vv) \vv}{R^2} - \frac{\vvd}{R} }\ret
	}
	
	By the chain rule,
	\eq{
		\dv{t} = \dv{t'}{t} \dv{t'}
		= \frac{1}{1 - \vbet(t') \vdot \nh(t')} \dv{t'}
		= \brac{ \frac{1}{1 - \vbet \vdot \nh} }\ret \dv{t'}.
	}
	For the second derivative, note that
	\al{
		\dv{t'}\!\brac{ \frac{1}{1 - \vbet \vdot \nh} }\ret &= \brac{ \dv{t}(\frac{1}{\vv \vdot \nh / c} - 1) }\ret
		= \brac{ \frac{1}{(1 - \vv \vdot \nh / c)^2} \dv{t}(1 - \frac{\nh \vdot \vv}{c}) }\ret \\
		&= \frac{1}{c} \brac{ \frac{1}{(1 - \vbet \vdot \nh)^2} \paren{ \nh \vdot \vvd + \frac{(\nh \vdot \vv)^2 - \vv^2}{R} } }\ret,
	}
	so
	\al{
		\dv[2]{t} &= \dv{t}(\frac{1}{1 - \vbet(t') \vdot \nh(t')} \dv{t'})
		= \frac{1}{1 - \vbet(t') \vdot \nh(t')} \dv{t'}(\frac{1}{1 - \vbet(t') \vdot \nh(t')} \dv{t'}) \\
		&= \frac{1}{1 - \vbet(t') \vdot \nh(t')} \curly{ \dv{t'}(\frac{1}{1 - \vbet(t') \vdot \nh(t')}) \dv{t'} + \frac{1}{1 - \vbet(t') \vdot \nh(t')} \dv[2]{{t'}} } \\
		&= \frac{1}{c} \brac{ \frac{1}{(1 - \vbet \vdot \nh)^3} \paren{ \nh \vdot \vvd + \frac{(\nh \vdot \vv)^2 - \vv^2}{R} } }\ret \dv{t'} + \brac{ \frac{1}{(1 - \vbet \vdot \nh)^2} }\ret \dv[2]{{t'}}.
	}
	
	The first term of Eq.~\refeq{HF} can be written
	\aln{
		e \brac{ \frac{\nh}{R^2} }\ret &= e \brac{ \frac{(1 - \vbet \vdot \nh)^3 \nh}{(1 - \vbet \vdot \nh)^3 R^2} }\ret
		= e \brac{ \frac{1}{(1 - \vbet \vdot \nh)^3 R^2} \paren{ \nh - 3 \frac{(\nh \vdot \vv) \nh}{c} + 3 \frac{(\nh \vdot \vv)^2 \nh}{c^2} - \frac{(\nh \vdot \vv)^3 \nh}{c^3} } }\ret \notag \\
		&= \frac{e}{c^3} \brac{ \frac{1}{(1 - \vbet \vdot \nh)^3 R^2} \paren{ c^3 \nh - 3 c^2 (\nh \vdot \vv) \nh + 3c (\nh \vdot \vv)^2 \nh - (\nh \vdot \vv)^3 \nh } }\ret. \label{term1}
	}
	
	The second term of Eq.~\refeq{HF} becomes
	\aln{
		e \brac{ \frac{R}{c} }\ret \dv{t} &\brac{ \frac{\nh}{R^2} }\ret = \frac{e}{c} \brac{ \frac{R}{1 - \vbet \vdot \nh} }\ret \dv{t'}\!\brac{ \frac{\nh}{R^2} }\ret
		= \frac{e}{c} \brac{ \frac{3 (\nh \vdot \vv) \nh - \vv}{(1 - \vbet \vdot \nh) R^2} }\ret \notag
%		&= \frac{e}{c} \brac{ \frac{1}{(1 - \vbet \vdot \nh) R^2} [ 3 (\nh \vdot \vv) \nh - \vv ] }\ret \notag
		= \frac{e}{c} \brac{ \frac{(1 - \vbet \vdot \nh)^2 \{ 3 (\nh \vdot \vv) \nh - \vv \}}{(1 - \vbet \vdot \nh)^3 R^2} }\ret \notag \\
		&= \frac{e}{c} \brac{ \frac{1}{(1 - \vbet \vdot \nh)^3 R^2} \paren{ 1 - 2 \frac{\nh \vdot \vv}{c} + \frac{(\nh \vdot \vv)^2}{c^2} } \{ 3 (\nh \vdot \vv) \nh - \vv \} }\ret \notag \\
		&= \frac{e}{c} \brac{ \frac{1}{(1 - \vbet \vdot \nh)^3 R^2} \paren{ 3 (\nh \vdot \vv) \nh - \vv - 6 \frac{(\nh \vdot \vv)^2 \nh}{c} + 2 \frac{ (\nh \vdot \vv)\vv}{c} + 3 \frac{(\nh \vdot \vv)^3 \nh}{c^2} - \frac{(\nh \vdot \vv)^2 \vv}{c^2} } }\ret \notag \\
		&= \frac{e}{c^3} \brac{ \frac{3 c^2 (\nh \vdot \vv) \nh - c^2 \vv - 6 c (\nh \vdot \vv)^2 \nh + 2 c (\nh \vdot \vv)\vv + 3 (\nh \vdot \vv)^3 \nh - (\nh \vdot \vv)^2 \vv}{(1 - \vbet \vdot \nh)^3 R^2} }\ret. \label{term2}
	}
	
	The third term of Eq.~\refeq{HF} becomes
	\eqn{thing2}{
		\frac{e}{c^2} \dv[2]{t} [ \nh ]\ret = \frac{e}{c^3} \brac{ \frac{1}{(1 - \vbet \vdot \nh)^3} \paren{ \nh \vdot \vvd + \frac{(\nh \vdot \vv)^2 - \vv^2}{R} } }\ret \dv{t'}[ \nh ]\ret + \frac{e}{c^2} \brac{ \frac{1}{(1 - \vbet \vdot \nh)^2} }\ret \dv[2]{{t'}} [ \nh ]\ret.
	}
	The first term of Eq.~\refeq{thing2} is
	\al{
		\frac{e}{c^3} &\brac{ \frac{1}{(1 - \vbet \vdot \nh)^3} \paren{ \nh \vdot \vvd + \frac{(\nh \vdot \vv)^2 - \vv^2}{R} } \frac{(\nh \vdot \vv) \nh - \vv}{R} }\ret \\
		&= \frac{e}{c^3} \brac{ \frac{1}{(1 - \vbet \vdot \nh)^3} \paren{ \frac{(\nh \vdot \vvd) (\nh \vdot \vv) \nh}{R} + \frac{(\nh \vdot \vv)^3 \nh}{R^2} - \frac{(\nh \vdot \vv) \vv^2 \nh}{R^2} - \frac{(\nh \vdot \vvd) \vv}{R} - \frac{(\nh \vdot \vv)^2 \vv}{R^2} + \frac{\vv^2 \vv}{R^2} } }\ret \\
		&= \frac{e}{c^3} \brac{ \frac{R (\nh \vdot \vvd) (\nh \vdot \vv) \nh + (\nh \vdot \vv)^3 \nh - (\nh \vdot \vv) \vv^2 \nh - R (\nh \vdot \vvd) \vv - (\nh \vdot \vv)^2 \vv + \vv^2 \vv}{(1 - \vbet \vdot \nh)^3 R^2} }\ret ,
	}
	and the second term of Eq.~\refeq{thing2} is
	\al{
		\frac{e}{c^2} &\brac{ \frac{1}{(1 - \vbet \vdot \nh)^2} \paren{ \frac{(\nh \vdot \vv) \nh}{R} + 3 \frac{(\nh \vdot \vv)^2 \nh}{R^2} - \frac{\vv^2 \nh}{R^2} - 2 \frac{(\nh \vdot \vv) \vv}{R^2} - \frac{\vvd}{R} } }\ret \\
		&= \frac{e}{c^2} \brac{ \frac{1 - \vbet \vdot \nh}{(1 - \vbet \vdot \nh)^3 R^2} \{ R (\nh \vdot \vv) \nh + 3 (\nh \vdot \vv)^2 \nh - \vv^2 \nh - 2 (\nh \vdot \vv) \vv - R \vvd \} }\ret \\
		&= \frac{e}{c^3} \bigg[ \frac{1}{(1 - \vbet \vdot \nh)^3 R^2} \{ R c (\nh \vdot \vv) \nh + 3 c (\nh \vdot \vv)^2 \nh - c \vv^2 \nh - 2 c (\nh \vdot \vv) \vv - R c \vvd \\
		&\phantom{mmmmmmmmmmmmmm} - R (\nh \vdot \vv)^2 \nh - 3 (\nh \vdot \vv)^3 \nh + (\nh \vdot \vv)\vv^2 \nh + 2 (\nh \vdot \vv)^2 \vv + R (\nh \vdot \vv) \vvd \} \bigg]\ret.
	}
	Summing the two terms, we find
	\aln{
	\frac{e}{c^2} \dv[2]{t} [ \nh ]\ret % &= \frac{e}{c^3} \bigg[ \frac{1}{(1 - \vbet \vdot \nh)^3 R^2} \bigg( \frac{\vv^2 (\nh \vdot \vv) \nh}{\gam} - \frac{R (\nh \vdot \vvd)(\nh \vdot \vv) \nh}{\gam} - (\nh \vdot \vv)^3 \nh - \frac{\vv^2 \vv}{\gam^2} + \frac{R (\nh \vdot \vvd) \vv}{\gam^2} + \frac{(\nh \vdot \vv)^2 \vv}{\gam} \\
%	&\phantom{mmmmmm} + 3 c (\nh \vdot \vv)^2 \nh - 2 \frac{c (\nh \vdot \vv) \vv}{\gam} + \frac{c R (\nh \vdot \vvd) \nh}{\gam} - \frac{c \vv^2 \nh}{\gam} - 2 \frac{c R \vvd}{\gam^2} + c R \vvd - 3 (\nh \vdot \vv)^3 \nh \\
%	&\phantom{mmmmmmmmmm} + 2 \frac{(\nh \vdot \vv)^2 \vv}{\gam} - \frac{R (\nh \vdot \vvd) (\nh \vdot \vv) \nh}{\gam} + \frac{\vv^2 (\nh \vdot \vv) \nh}{\gam} + 2 \frac{R (\nh \vdot \vv) \vvd}{\gam^2} - R (\nh \vdot \vv) \vvd \bigg) \bigg]\ret \\
%	%
	&= \frac{e}{c^3} \bigg[ \frac{1}{(1 - \vbet \vdot \nh)^3 R^2} \bigg( 2 \frac{\vv^2 (\nh \vdot \vv) \nh}{\gam} - 2\frac{R (\nh \vdot \vvd)(\nh \vdot \vv) \nh}{\gam} - 4 (\nh \vdot \vv)^3 \nh - \frac{\vv^2 \vv}{\gam^2} + \frac{R (\nh \vdot \vvd) \vv}{\gam^2} \notag \\
	&\phantom{mmmmmmmmmmmmmm} + 3 \frac{(\nh \vdot \vv)^2 \vv}{\gam} + 3 c (\nh \vdot \vv)^2 \nh - 2 \frac{c (\nh \vdot \vv) \vv}{\gam} + \frac{c R (\nh \vdot \vvd) \nh}{\gam} - \frac{c \vv^2 \nh}{\gam} \notag \\
	&\phantom{mmmmmmmmmmmmmmmmmm} - 2 \frac{c R \vvd}{\gam^2} + c R \vvd + 2 \frac{R (\nh \vdot \vv) \vvd}{\gam^2} - R (\nh \vdot \vv) \vvd \bigg) \bigg]\ret. \label{term3}
	}
	
	Summing Eqs.~\refeq{term1} and \refeq{term2},
	\aln{
		e \brac{ \frac{\nh}{R^2} }\ret &+ e \brac{ \frac{R}{c} }\ret \dv{t} \brac{ \frac{\nh}{R^2} }\ret \notag \\
		&= \frac{e}{c^3} \brac{ \frac{1}{(1 - \vbet \vdot \nh)^3 R^2} \paren{ c^3 \nh - 3c (\nh \vdot \vv)^2 \nh + 2 (\nh \vdot \vv)^3 \nh - \frac{c^2 \vv}{\gam} + 2 \frac{c (\nh \vdot \vv)\vv}{\gam} - \frac{(\nh \vdot \vv)^2 \vv}{\gam} } }\ret. \label{term12}
	}
	Summing Eqs.~\refeq{term12} and \refeq{term3},
	\al{
		\vE &= \frac{e}{c^3} \bigg[ \frac{1}{(1 - \vbet \vdot \nh)^3 R^2} \bigg( c^3 \nh - 2 (\nh \vdot \vv)^3 \nh - \frac{c^2 \vv}{\gam} + 2 \frac{(\nh \vdot \vv)^2 \vv}{\gam} + 2 \frac{\vv^2 (\nh \vdot \vv) \nh}{\gam} - 2\frac{R (\nh \vdot \vvd)(\nh \vdot \vv) \nh}{\gam} - \frac{\vv^2 \vv}{\gam^2} \\
	&\phantom{mmmmmmmmmmmm} + \frac{R (\nh \vdot \vvd) \vv}{\gam^2} + \frac{c R (\nh \vdot \vvd) \nh}{\gam} - \frac{c \vv^2 \nh}{\gam} - 2 \frac{c R \vvd}{\gam^2} + c R \vvd + 2 \frac{R (\nh \vdot \vv) \vvd}{\gam^2} - R (\nh \vdot \vv) \vvd \bigg) \bigg]\ret
	}
}
