\state{(Jackson 9.4)}{
	Apply the approach of Prob.~{1(b)} to the current and magnetization densities of the particle of charge $q$ rotating about the origin in the $xy$ plane in a circle of radius $R$ at constant angular speed $\omgo$.  The motion is such that $\omgo R \ll c$.
}

%
%	Jackson 9.4(a)
%

\prob{}{
	Find $\Jxn$, $\Jyn$, and $\Jzn$ in terms of cylindrical coordinates for all $n$.  Also determine the components of the orbital ``magnetization,'' $(\vx \cross \vJn) / 2$, and its divergence (which plays the role of a magnetic charge density for magnetic multipoles, as in
	\eqn{Mlm}{
		\Mlm = -\frac{1}{l + 1} \int r^l \,\Ylm^*(\tht, \phi) \,\grad \vdot (\vx \vdot \vJ) \ddcx ).
	}
	\vfix
}

\sol{
	The current density is given by $\vJ = \rho \vv$, where $\vv$ is the velocity of the particle.  Using Eq.~\refeq{ptch},
	\eq{
		\vv = \dv{\vx(t)}{t}
		= \dv{t}[ R \cos(\omgo t) \,\xh + R \sin(\omgo t) \,\yh ]
		= -R \omgo \sin(\omgo t) \,\xh + R \omg \cos(\omgo t) \,\yh
		= R \omgo \Re[ i e^{i \omgo t} \,\xh + e^{i \omgo t} \,\yh ].
	}
	Using the approach of Prob.~{1(b)} and adapting the right side of Eq.~\refeq{decomp},
	\al{
		\vJn(\vx) &= \frac{1}{T} \intoT \vJ(\vx, t) \,e^{i n \omgo t} \ddt
		= \frac{1}{T} \intoT \rho(\vx, t) \,\vv(t) \,e^{i n \omgo t} \ddt \\
%		&= \frac{q \omgo}{R T} \intoT \del(r - R) \,\del(\phi - \omgo t) \,\del(z) [ i e^{i \omgo t} \,\xh + e^{i \omgo t} \,\yh ] \,e^{i n \omgo t} \ddt \\
		&= \frac{1}{2\pi} \frac{q \omgo^2}{R} \intoT \del(r - R) \,\del(\phi - \omgo t) \,\del(z) \,e^{i (n + 1) \omgo t} [ i \,\xh + \yh ] \ddt
		= \frac{1}{2\pi} \frac{q \omgo^2}{R} \del(r - R) \,\del(\phi - \omgo t) \,\del(z) \,e^{i (n + 1) \phi} [ i \,\xh + \yh ]
	}
	where $\rho(\vx, t)$ is given by Eq.~\refeq{rhocy}.  So the components are
	\ans{ \al{
		\Jxn &= \frac{i}{2\pi} \frac{q \omgo^2}{R} \del(r - R) \,\del(\phi - \omgo t) \,\del(z) \,e^{i (n + 1) \phi}
		= -\frac{1}{2\pi} \frac{q \omgo^2}{R} \del(r - R) \,\del(\phi - \omgo t) \,\del(z) \,e^{i n \phi} \sin\phi, \\
		\Jyn &= \frac{1}{2\pi} \frac{q \omgo^2}{R} \del(r - R) \,\del(\phi - \omgo t) \,\del(z) \,e^{i (n + 1) \phi}
		= \frac{1}{2\pi} \frac{q \omgo^2}{R} \del(r - R) \,\del(\phi - \omgo t) \,\del(z) \,e^{i n \phi} \cos\phi, \\
		\Jzn &= 0.
	} }%
	
	Let $\vMn = (\vx \cross \vJn) / 2$, and note that
	\al{
		\vMn &= \frac{[ y \,\Jzn - z \,\Jyn ] \,\xh + [ z \,\Jxn - x \,\Jzn ] \,\yh + [ x \,\Jyn - y \,\Jxn ] \,\zh}{2} \\
		&= \frac{1}{4\pi} \frac{q \omgo^2}{R} \del(r - R) \,\del(z) \,e^{i (n + 1) \phi} [ -z \,\xh + i z \,\yh + (x - i y) \,\zh ].
	}
	Transforming to cylindrical coordinates, in which $x - i y = r \cos\phi - i r \sin\phi = r e^{-i\phi}$, the components are
	\ans{ \al{
		\Mxn &= -\frac{1}{4\pi} \frac{q \omgo^2}{R} \del(r - R) \,z \,\del(z) \,e^{i (n + 1) \phi}
		= -\frac{1}{4\pi} \frac{q \omgo^2}{R} \del(r - R) \,z \,\del(z) \,e^{i n \phi} \cos\phi, \\
		\Myn &= \frac{i}{4\pi} \frac{q \omgo^2}{R} \del(r - R) \,z \,\del(z) \,e^{i (n + 1) \phi}
		= -\frac{1}{4\pi} \frac{q \omgo^2}{R} \del(r - R) \,z \,\del(z) \,e^{i n \phi} \sin\phi, \\
		\Mzn &= \frac{1}{4\pi} \frac{q \omgo^2}{R} r \,\del(r - R) \,\del(z) \,e^{i n \phi}.
	} }%

	To find the divergence, we will write $\vM$ in cylindrical coordinates.  According to the inside cover of Griffiths, $\rh = \cos\phi \,\xh + \sin\phi \,\yh$.  So we have
	\eq{
		\vMn = -\frac{1}{4\pi} \frac{q \omgo^2}{R} \del(r - R) \,\del(z) \,e^{i n \phi} (z \,\cos\phi \,\xh + z \,\sin\phi \,\yh + r \,\zh)
		= -\frac{1}{4\pi} \frac{q \omgo^2}{R} \del(r - R) \,\del(z) \,e^{i n \phi} (z \,\rh - r \,\zh).
	}
%	\al{
%		\xh &= \cos\phi \,\rh - \sin\phi \,\phh, &
%		\yh &= \sin\phi \,\rh + \cos\phi \,\phh, &
%		\zh &= \zh.
%	}
%	Then $\vMn$ becomes
%	\al{
%		\vMn &= \frac{1}{4\pi} \frac{q \omgo^2}{R} \del(r - R) \,\del(z) \,e^{i (n + 1) \phi} [ -z (\cos\phi \,\rh - \sin\phi \,\phh) + i z (\sin\phi \,\rh + \cos\phi \,\phh) + r (\cos\phi - i \sin\phi) \,\zh ] \\
%		&= \frac{1}{4\pi} \frac{q \omgo^2}{R} \del(r - R) \,\del(z) \,e^{i (n + 1) \phi} [ z (-\cos\phi + i \sin\phi) \,\rh + z (i \cos\phi + \sin\phi) \,\phh + r (\cos\phi - i \sin\phi) \,\zh ] \\
%		&= \frac{1}{4\pi} \frac{q \omgo^2}{R} \del(r - R) \,\del(z) \,e^{i (n + 1) \phi} [ -z (\cos\phi - i \sin\phi) \,\rh + i z (\cos\phi - i \sin\phi) \,\phh + r (\cos\phi - i \sin\phi) \,\zh ] \\
%		&= \frac{1}{4\pi} \frac{q \omgo^2}{R} \del(r - R) \,\del(z) \,e^{i n \phi} [ -z \,\rh + i z \,\phh + r \,\zh ].
%	}
	The divergence in cylindrical coordinates is on the inside cover of Jackson:
	\eq{
		\grad \vdot \vA = \frac{1}{r} \pdv{r}(r A_r) + \frac{1}{r} \pdv{\phi}{A_\phi} + \pdv{A_z}{z}.
	}
	Note that
	\al{
		\pdv{r}[ r (M_r)_n) ] &\propto \pdv{r}[ r \,\del(r - R) ]
		= r \pdv{r}[ \del(r - R) ] + \del(r - R)
		= \del(r - R) - \del(r - R)
		= 0, \\
%		\pdv{\phi}[ (M_\phi)_n ] &\propto i \pdv{\phi}(e^{i (n - 1) \phi})
%		= i^2 (n - 1) \,e^{i n \phi}
%		= -n e^{i n \phi}, \\
		\pdv{z}[ \Mzn ] &\propto \pdv{z}[ \del(z) ]
		= -\del(z) \pdv{z},
	}
	where we have used $\del'(x) = -\del(x) \dv*{x}$ and $x \,\del'(x) = -\del(x)$~\cite{Delta}.  So the divergence is
	\eqn{div}{
		\ans{ \grad \vdot \vMn = -\frac{1}{4\pi} \frac{q \omgo^2}{R} r \, \del(r - R) \,e^{i n \phi} \,\del(z) \pdv{z} }.
	}
%	\eqn{div}{
%		\ans{ \grad \vdot \vMn = -\frac{1}{4\pi} \frac{q \omgo^2}{R} \del(r - R) \,\del(z) \,e^{i n \phi} \paren{ i n \frac{z}{r} + r \pdv{z} }. }
%	}
	\vfix
}

%
%	Jackson 9.4(b)
%

\prob{}{
	What long-wavelength magnetic multipoles $(l, m)$ occur and at what frequencies?  (Remember that the multipole order $l$ does not necessarily equal the harmonic number $n$.)
}

\sol{
	The long-wavelength magnetic multipole moments are given by Eq.~\refeq{Mlm}.  (This system has no intrinsic magnetization, so this is the only relevant equation.)  Writing Eq.~\refeq{div}, in spherical coordinates with
	\al{
		r &\to r \sin\tht, &
		z &\to r \cos\tht, &
		\pdv{z} &\to \cos\tht \pdv{r} - \frac{\sin\tht}{r} \pdv{\tht},
	}
	we have
	\al{
		\grad \vdot \vMn = \frac{1}{4\pi} \frac{q \omgo^2}{R} \del(r - R) \,\del(\tht - \pi/2) \,e^{i n \phi} \sin^2\tht \pdv{\tht}.
	}
	Making this substitution in Eq.~\refeq{Mlm}, and noting that
	\eq{
		\sin\tht \dd{\tht} = -\dd{(\cos\tht)}
		\qimplies
		\frac{1}{\sin\tht} \pdv{\tht} = -\pdv{}{(\cos\tht)},
	}
	we find
	\al{
		\Mlm &= -\frac{1}{l + 1} \int \grad \vdot \vMn \,r^l \,\Ylm^*(\tht, \phi) \ddcx \\
		&= \frac{1}{4\pi (l + 1)} \frac{q \omgo^2}{R} \int \del(r - R) \,\del(\tht - \pi/2) \,e^{i n \phi} \brac{ r \sin\tht \cos\tht \pdv{r} - \sin^2\tht \pdv{\tht} } r^l \,\Ylm^*(\tht, \phi) \ddcx \\
		&= \frac{1}{4\pi (l + 1)} \frac{q \omgo^2}{R} \int \del(r - R) \,\del(\tht - \pi/2) \,e^{i n \phi} \brac{ r \sin\tht \cos\tht \pdv{r} - \sin^2\tht \pdv{\tht} } r^l \,\Ylm^*(\tht, \phi) \ddcx \\
		&= \frac{1}{4\pi (l + 1)} \frac{q \omgo^2}{R} \bigg[
			l \int \del(r - R) \,\del(\tht - \pi/2) \,e^{i n \phi} \sin\tht \cos\tht \,r^l \,\Ylm^*(\tht, \phi) \ddcx \\
			&\phantom{mmmmmmmmmmmmmmmmmmmmmmmm} + \int \del(r - R) \,\del(\tht - \pi/2) \,e^{i n \phi} \sin^3\tht \,r^l \pdv{\Ylm^*(\tht, \phi)}{(\cos\tht)} \ddcx \bigg].
	}
%	\al{
%		\grad \vdot \vMn = -\frac{1}{4\pi} \frac{q \omgo^2}{R} \del(r - R) \,\del(\tht - \pi/2) \,e^{i n \phi} \paren{ i n \frac{\cos\tht}{\sin\tht} + r \sin\tht \cos\tht \pdv{r} - \sin^2\tht \pdv{\tht} }.
%	}
%	Making this substitution in Eq.~\refeq{Mlm}, and noting that
%	\eq{
%		\sin\tht \dd{\tht} = -\dd{(\cos\tht)}
%		\qimplies
%		\frac{1}{\sin\tht} \pdv{\tht} = -\pdv{}{(\cos\tht)},
%	}
%	we find
%	\al{
%		\Mlm &= -\frac{1}{l + 1} \int \grad \vdot \vMn \,r^l \,\Ylm^*(\tht, \phi) \ddcx \\
%		&= \frac{1}{4\pi (l + 1)} \frac{q \omgo^2}{R} \int \del(r - R) \,\del(\tht - \pi/2) \,e^{i n \phi} \brac{ i n \frac{\cos\tht}{\sin\tht} + r \sin\tht \cos\tht \pdv{r} - \sin^2\tht \pdv{\tht} } r^l \,\Ylm^*(\tht, \phi) \ddcx \\
%		&= \frac{1}{4\pi (l + 1)} \frac{q \omgo^2}{R} \bigg[
%			i n \int \del(r - R) \,\del(\tht - \pi/2) \,e^{i n \phi} \frac{\cos\tht}{\sin\tht} r^l \,\Ylm^*(\tht, \phi) \ddcx \\
%			&\phantom{mmmmmmmmmmmmmmmm} + l \int \del(r - R) \,\del(\tht - \pi/2) \,e^{i n \phi} \sin\tht \cos\tht \,r^l \,\Ylm^*(\tht, \phi) \ddcx \\
%			&\phantom{mmmmmmmmmmmmmmmmmmmmmmmm} + \int \del(r - R) \,\del(\tht - \pi/2) \,e^{i n \phi} \sin^3\tht \,r^l \pdv{\Ylm^*(\tht, \phi)}{(\cos\tht)} \ddcx \bigg].
%	}
%	\clearpage
	Since the spherical harmonics are mutually orthogonal, all of the integrals will be nonzero only for $m = n$.  By inspection, we can also associate the first integral with $Y_{21}$ and the third with $Y_{43}$, where
	\al{
		Y_{21}(\tht, \phi) &= -\sqrt{\frac{15}{8\pi}} \sin\tht \cos\tht \,e^{i\phi}, &
		Y_{43}(\tht, \phi) &= -\frac{3}{8} \sqrt{\frac{35}{\pi}} \cos\tht \sin^3\tht \,e^{3i\phi}.
	}
	We know from Prob.~{1} that the radiation frequencies are $m \omgo$.  This means that \ans{the long-wavelength magnetic multipoles $(2, 1)$ and $(4, 3)$ occur with respective frequencies $\omgo$ and $3\omgo$.}
}
