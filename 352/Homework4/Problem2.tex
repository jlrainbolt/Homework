\state{Pauli paramagnetism}{
	Cold atomic gases could be realized by atomic isotopes which are fermions ($^6$Li, $^{40}$K, etc.).  Such isotopes may have a large atomic spin.  Assuming that the Fermi gas is degenerate and its constituents have a spin $s > 1 / 2$, compute the Pauli magnetic susceptibility.
}

\sol{
	The atoms gain additional spin energy in the presence of a magnetic field $\vB = B \,\zh$.  In the spin-$1/2$ case, the thermodynamic potential becomes
	\eqn{thingy}{
		\Omg(\mu) = \frac{\Omgo(\mu + \muB B)}{2} + \frac{\Omgo(\mu - \muB B)}{2},
	}
	where $\Omgo$ is the thermodynamic potential when no magnetic field is present and $\muB$ is the Bohr magneton.  This formulation is due to each particle's picking up extra energy $\pm \muB B$ from the component of its spin in the direction of the field.  Since the thermodynamic potential depends on $\eps - \mu$, we can equivalently make the substitution $\mu \to \mu \mp \muB B$~\cite[p.~172]{Landau2}.
	
	For an atom of arbitrary spin $s$, there are $2s + 1$ possible $z$ components of the spin.  They are described by the matrix $\Sz$.  Note that~\cite[p.~375]{Shankar}
	\al{
		\Sz &= \frac{\hbar}{2} \mqty[ 1 & 0 \\ 0 & -1 ]
		= \hbar \mqty[ s & 0 \\ 0 & -s ]
		\quad (s = 1/2), &
		\Sz &= \hbar \mqty[
			s & 0 & 0 & \cdots & 0 \\
			0 & s - 1 & 0 & \cdots & 0 \\
			0 & 0 & s - 2 & \cdots & 0 \\
			\vdots & \vdots & \vdots & \ddots & \vdots \\
			0 & 0 & 0 & \cdots & -s ]
			\quad (s \text{ arbitrary}).
	}
	Then, by analogy with Eq.~\refeq{thingy}, for arbitrary $s$
	\eqn{thing2}{
		\Omg(\mu) = \frac{1}{2s + 1} \sumjs \Omgo[ \mu + 2(s - j) \muB B ]
		\equiv \frac{1}{2s + 1} \sumjs \Omgo(\muj),
	}
	where we have defined $\muj = \mu + 2(s - j) \muB B$.
	
	Since $B$ is small, we can Taylor expand Eq.~\refeq{thing2} about $\muB B = 0$~\cite[p.~172]{Landau2}:
	\eq{
		\Omg(\mu) \approx \bigg[ \Omg(\mu) \bigg]_{\muB B = 0} + \muB B \brac{ \pdv{\Omg(\mu)}{(\muB B)} }_{\muB B = 0} + \frac{\muB^2 B^2}{2} \brac{ \pdv[2]{\Omg(\mu)}{(\muB B)} }_{\muB B = 0}.
	}
	Note that
	\al{
		\pdv{\Omg}{(\muB B)} &= \frac{1}{2s + 1} \sumjws \pdv{\Omgo(\mu)}{\muj} \pdv{\muj}{(\muB B)}
		= \frac{1}{2s + 1} \sumjws 2(s - j) \pdv{\Omgo(\mu)}{\muj}, \\
		\pdv[2]{\Omg}{(\muB B)} &= \frac{1}{2s + 1} \sumjws 2(s - j) \pdv[2]{\Omgo(\mu)}{\muj} \pdv{\muj}{(\muB B)}
		= \frac{1}{2s + 1} \sumjws 4 (s - j)^2 \pdv[2]{\Omgo(\mu)}{\muj},
	}
	so
	\al{
		\brac{ \pdv{\Omg}{(\muB B)} }_{\muB B = 0} &= \frac{1}{2s + 1} \sumjws 2(s - j) \brac{ \pdv{\Omgo(\mu)}{\muj} }_{\muB B = 0}
		= \frac{1}{2s + 1} \sumjs 2(s - j) \pdv{\Omgo(\mu)}{\mu}
		= 0, \\[1.5ex]
		\brac{ \pdv[2]{\Omg}{(\muB B)} }_{\muB B = 0} &= \frac{1}{2s + 1} \sumjws 4 (s - j)^2 \brac{ \pdv[2]{\Omgo(\mu)}{\muj} }_{\muB B = 0}
		= \frac{1}{2s + 1} \sumjws 4 (s - j)^2 \pdv[2]{\Omgo(\mu)}{\mu} \\
		&= \frac{8}{2s + 1} \pdv[2]{\Omgo(\mu)}{\mu} \sumjs (s - j)^2,
	}
	where we have used the fact that $s$ is an integer multiple of $1/2$ for a fermion.  So we have
	\eq{
		\Omg(\mu) \approx \Omgo(\mu) + \frac{4 \muB^2 B^2}{2s + 1} \pdv[2]{\Omgo(\mu)}{\mu} \sumjs (s - j)^2.
	}
	
	The magnetic moment of the gas is $M = -(\pdv*{\Omg}{B})_{T, V, \mu}$~\cite[p.~172]{Landau2}.  Here,
	\eq{
		M = -\frac{8 \muB^2 B}{2s + 1} \pdv[2]{\Omgo(\mu)}{\mu} \sumjs (s - j)^2.
	}
	According to p.~2 of lecture 12, the paramagnetic susceptibility is defined $\chi = (\pdv*{M}{B}) / V$.  Then
	\eq{
		\chi = -\frac{8 \muB^2}{(2s + 1) V} \pdv[2]{\Omgo(\mu)}{\mu} \sumjs (s - j)^2
		= \frac{8 \muB^2}{(2s + 1) V} \paren{ \pdv{N}{\mu} }_{T, V} \sumjs (s - j)^2,
	}
	where we have used $\pdv*{\Omgo}{\mu} = -N$~\cite[p.~172]{Landau2}.  The number of particles in a degenerate Fermi gas is
	\eq{
		N = \frac{V}{3\pi^2 \hbar^3} (2 m \mu)^{3/2},
	}
	where $m$ is the mass of each atom~\cite[p.~173]{Landau2}.  So
	\al{
		\chi &= \frac{8 \muB^2}{(2s + 1) V} \pdv{\mu}(\frac{V}{3\pi^2 \hbar^3} (2 m \mu)^{3/2}) \sumjs (s - j)^2
		= \frac{3}{2} \frac{8 \muB^2}{(2s + 1) V} \frac{V}{3\pi^2 \hbar^3} \sqrt{2^3 m^3 \mu} \sumjs (s - j)^2 \\
		&= \ans{ \frac{8 \muB^2}{\pi^2 \hbar^3 (2s + 1)}  \sqrt{2 m^3 \mu} \sumjs (s - j)^2. }
	}
}
