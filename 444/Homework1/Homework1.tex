\state{Alternative regulators in QED (Peskin \& Schroeder 7.2)}{
	In Section.~7.5, we saw that the Ward identity can be violated by an improperly chosen regulator.  Let us check the validity of the identity $\Zq = \Zw$, to order $\alp$, for several choices of the regulator.  We have already verified that the relation holds for Paul-Villars regularization.
}

\prob{
	Recompute $\del\Zq$ and $\del\Zw$, defining the integrals~(6.49) and (6.50) by simply placing an upper limit $\Lam$ on the integration over $\ellE$.  Show that, with this definition, $\del\Zq \neq \del\Zw$.
}

\sol{
	From (7.47) in Peskin \& Schroeder,
	\eqn{7.47}{
		\Gammqo = \frac{1}{\Zq} \gamm,
	}
	we can find an expression for $\del\Zq$, similar to how (7.31) is obtained from (7.26).  Roughly,
	\eqn{thingy}{
		\frac{1}{\Zq + \del\Zq} \gamm \approx \Zq (1 - \del\Zq) \gamm
		= \Gammqo + \del\Gammqo
		\qimplies
		\del\Gammqo = -\del\Zq \gamm.
	}
	According to (6.33),
	\eq{
		\Gammppp = \gamm \Fq(q^2) + \frac{i \sigmn \qsn}{2 m} \Fw(q^2).
	}
	We note that $\Gamm = \gamm$, $\Fq = 1$, and $\Fw = 0$ to lowest order~\cite[pp.~185--186]{Peskin}.  Then we can write
	\eqn{Gam}{
		\del\Gammqo = \gamm \del\Fqo + \frac{i \sigmn \qsn}{2 m} \del\Fwo.
	}
	Using this equation and the identity $\gamm \gamsm = 4$~\cite{Gamma}, Eq.~\refeq{thingy} can be written
	\eqn{Z1}{
		\del \Zq = -\frac{1}{4} \gamsm \del\Gammqo
		= -\del\Fqo - \gamsm \frac{i \sigmn \qsn}{8 m} \del\Fwo.
	}
	
	In order to find $\del\Gamm$ we use (6.47):
	\aln{
		\ubpp \del\Gammppp \up &= 2 i e^2 \int \ddqlf \intoq \ddx \ddy \ddz \del(x + y + z - 1) \frac{2}{D^3} \notag \\
		&\hspace{5em} \phantom{=\ } \times \ubpp \left\{ \gamm \brac{ -\frac{\ell^2}{2} + (1 - x) (1 - y) q^2 + (1 - 4z + z^2) m^2 } \right. \label{thing1} \\
		&\hspace{15em} \phantom{=\ } \left. + \frac{i \sigmn \qsn}{2 m} [ 2 m^2 z (1 - z) ] \right\} \up, \notag
	}
	where $\Del \equiv -x y q^2 + (1 - z)^2 m^2$ by (6.44), $\ell \equiv k + y q - z p$, and $D = \ell^2 - \Del + i \eps$~\cite[p.~191]{Peskin}.  The momenta $k$ and $p$ are assigned in the Feynman diagram on p.~189 of Peskin \& Schroeder, and $x, y$ are Feynman parameters~\cite[p.~190]{Peskin}.

	The forms of the two integrals we need to compute are given by Peskin \& Schroeder~(6.49) and (6.50).  Equation~(6.49) is
	\eqn{6.49}{
		\int \ddqlf \frac{1}{(\ell^2 - \Delta)^m} = \frac{i (-1)^m}{(4\pi)^2} \frac{1}{(m - 1) (m - 2)} \frac{1}{\Delta^{m - 2}}.
	}
	Here $m = 3$ because we have $D^{-3}$ in Eq.~\refeq{thing1}.  We can evaluate the left-hand side using the Euclidian 4-momentum defined in (6.48),
	\aln{ \label{ellE}
		\ello &\equiv i \ellEo, &
		\vell &= \vellE.
	}
	Following the steps on p.~193, we have
	\eq{
		\int \ddqlf \frac{1}{(\ell^2 - \Delta)^3} = \frac{i}{(-1)^3} \frac{1}{(2\pi)^4} \int \frac{\ddqlE}{(\ellE^2 + \Del)^3}
		= \frac{i (-1)^3}{(2\pi)^4} \int \ddOmgq \intoLam \ddlE \frac{\ellE^3}{(\ellE^2 + \Delta)^3},
	}
	where we have replaced the upper (infinite) bound of integration by a finite number $\Lam$.  Evaluating this integral using Mathematica and using $\int \ddOmgq = 2 \pi^2$~\cite[p.~193]{Peskin}, we find
	\aln{
		\int \ddqlf \frac{1}{(\ell^2 - \Del)^3} &= \frac{i (-1)^3}{(2\pi)^4} (2 \pi^2) \frac{\Lam^4}{4 \Del (\Del + \Lam^2)^2} \notag \\
		&= -\frac{i}{32 \pi^2} \frac{\Lam^4}{\Del (\Del + \Lam^2)^2} \notag \\
		&\approx -\frac{i}{32 \pi^2} \frac{1}{\Del}
		\equiv \alp, \label{ans1}
	}
	where we have taken the limit $\Lam \gg \Del$~\cite[p.~218]{Peskin} and defined $\alp$.  Equation~(6.50) in Peskin \& Schroeder is
	\eq{
		\int \ddqlf \frac{\ell^2}{(\ell^2 - \Del)^m} = \frac{i (-1)^{m - 1}}{(4\pi)^2} \frac{2}{(m - 1) (m - 2) (m - 3)} \frac{1}{\Del^{m - 3}}.
	}
	Following similar steps as for Eq.~\refeq{thing1}, the left-hand side is
	\aln{
		\int \ddqlf \frac{\ell^2}{(\ell^2 - \Del)^3} &= \frac{i}{(-1)^3} \frac{1}{(2\pi)^4} \int \ddqlE \frac{\ellE^2}{(\ellE^2 + \Del)^3} \notag \\
		&= \frac{i (-1)^3}{(2\pi)^4} \int \ddOmgq \intoLam \ddlE \frac{\ellE^5}{(\ellE^2 + \Del)^3} \notag \\
		&= \frac{i (-1)^3}{(2\pi)^4} (2 \pi^2) \frac{1}{4} \brac{ \frac{\Del (3 \Del + 4 \Lam^2)}{(\Del + \Lam^2)^2} + 2 \ln(\frac{\Del + \Lam^2}{\Del}) - 3 } \notag \\
		&= -\frac{i}{32 \pi^2} \brac{ \frac{\Del (3 \Del + 4 \Lam^2)}{(\Del + \Lam^2)^2} + 2 \ln(\frac{\Del +\Lam^2}{\Del}) - 3 } \notag \\
		&\approx -\frac{i}{16 \pi^2} \brac{ \ln(\frac{\Del +\Lam^2}{\Del}) - \frac{3}{2} }
		\equiv \bet, \label{ans2}
	}
	where we have defined $\bet$ and ignored terms of $\order{\Lam^{-2}}$~\cite[p.~218]{Peskin}. % We also ignore constant terms since they do not diverge~\cite[p.~196]{Peskin}.
	
	We now set $q^2 = 0$, and define $\Delo = (1 - z)^2 m^2$.  Then $\Del \to \Delo$ in our expression and $\alp \to \alpo, \bet \to \beto$ (which are functions of $\Delo$).  Feeding in Eqs.~\refeq{ans1} and \refeq{ans2}, Eq.~\refeq{thing1} can be written
	\eq{
		\ubpp \del\Gammqo \up = 2 i e^2 \intoq \ddx \ddy \ddz \del(x + y + z - 1) \ubpp \int \curly{ \gamm \brac{ -\beto + 2 (1 - 4z + z^2) m^2 \alpo } } \up.
	}
	Then
	\al{
		\del\Fqo &= 2 i e^2 \intoq \ddx \ddy \ddz \del(x + y + z - 1) \brac{ -\beto + 2 (1 - 4z + z^2) m^2 \alpo } \\
		&= 2 i e^2 \intoq \ddz (1 - z) \brac{ -\beto + 2 m^2 (1 - 4z + z^2) \alpo }, \\[2ex]
		%
		\del\Fwo &= 8 i e^2 \intoq \ddx \ddy \ddz \del(x + y + z - 1) m^2 z (1 - z) \alpo \\
		&= 8 i e^2 \intoq \ddz m^2 z (1 - z)^2 \alpo.
	}
	We ignore $\del\Fwo$ since it is not affected by the divergence~\cite[p.~196]{Peskin}.  In order to avoid issues coming from the divergence in $\del\Fqo$, we add a $z \mu^2$ term to $\Delo$~\cite[p.~195]{Peskin}.  So, feeding these results into Eq.~\refeq{Z1}, we obtain
	\aln{
		\del\Zq &= -2 i e^2 \intoq \ddz (1 - z) \brac{ -\beto + 2 (1 - 4z + z^2) m^2 \alpo } \notag \\
		&= -2 i e^2 \intoq \ddz (1 - z) \curly{ \frac{i}{16 \pi^2} \brac{ \ln(\frac{\Delo +\Lam^2}{\Delo}) - \frac{3}{2} } - 2 (1 - 4z + z^2) m^2 \frac{i}{32 \pi^2} \frac{1}{\Delo} } \notag \\
		&= \frac{e^2}{8 \pi^2} \intoq \ddz (1 - z) \brac{ \ln(\frac{\Delo +\Lam^2}{\Delo}) - \frac{3}{2} - \frac{m^2 (1 - 4z + z^2)}{\Delo} }, \label{Z1ans}
	}
	where
	\eqn{Delo}{
		\Delo = (1 - z)^2 m^2 + z \mu^2.
	}
	
	For $\del\Zw$, we can use the first part of Peskin \& Schroeder (7.31),
	\eqn{Z2}{
		\del\Zw = \left. \dv{\Sigw}{\psl} \right|_{\psl = m},
	}
	where $\Sigw$ is given by (7.17),
	\eqn{Sigw}{
		-i \Sigwp = -e^2 \intoq \ddx \int \ddqlf \frac{-2 x \psl + 4 \mo}{(\ell^2 - \Del + i \eps)^2},
	}
	where $\Del = -x (1 - x) p^2 + x \mu^2 + (1 - x) \mo^2$.  We may once again follow the steps on p.~193 to evaluate the integral, now with $m = 2$.  Changing the upper bound of integration to $\Lam$ once more, we have
	\al{
		\int \ddqlf \frac{1}{(\ell^2 - \Delta)^2} &= \frac{i}{(-1)^2} \frac{1}{(2\pi)^4} \int \ddqlE \frac{1}{(\ellE^2 + \Del)^2} \\
		&= \frac{i (-1)^2}{(2\pi)^4} \int \ddOmgq \intoLam \ddlE \frac{\ellE^3}{(\ellE^2 + \Del)^2} \\
		&= \frac{i}{(2\pi)^4} (2 \pi^2) \frac{1}{2} \brac{ \frac{\Del}{\Del + \Lam^2} + \ln(\frac{\Del + \Lam^2}{\Del}) - 1 } \\
		&= \frac{i}{16 \pi^2} \brac{ \frac{\Del}{\Del + \Lam^2} + \ln(\frac{\Del + \Lam^2}{\Del}) - 1 } \\
		&\approx \frac{i}{16 \pi^2} \brac{ \ln(\frac{\Del + \Lam^2}{\Del}) - 1 },
	}
	where we have evaluated the integral using Mathematica and dropped terms of $\order{\Lam^{-2}}$.  Substituting back into Eq.~\refeq{Sigw}, we find
	\eq{
		\Sigwp = -i e^2 \intoq \ddx (-2 x \psl + 4 \mo) \frac{i}{16 \pi^2} \brac{ \ln(\frac{\Del + \Lam^2}{\Del}) - 1 }.
	}
	Note that
	\aln{
		\dv{\Sigw}{\psl} &= \frac{e^2}{16 \pi^2} \dv{\psl} \curly{ \intoq \ddx (-2 x \psl + 4 \mo) \brac{ \ln(\frac{\Del + \Lam^2}{\Del}) - 1 } } \notag \\
		&= \frac{e^2}{16 \pi^2} \intoq \ddx \curly{ \brac{ \ln(\frac{\Del + \Lam^2}{\Del}) - 1 } \dv{\psl}(-2 x \psl + 4 \mo) + (-2 x \psl + 4 \mo) \dv{\psl} \brac{ \ln(\frac{\Del + \Lam^2}{\Del}) - 1 } } \notag \\
		&= \frac{e^2}{16 \pi^2} \intoq \ddx \curly{ \brac{ \ln(\frac{\Del + \Lam^2}{\Del}) - 1 } \dv{\psl}(-2 x \psl + 4 \mo) + (-2 x \psl + 4 \mo) \dv{\Del} \brac{ \ln(\frac{\Del + \Lam^2}{\Del}) - 1 } \dv{\Del}{\psl} }. \label{der}
	}
	Using $p^2 = \psl^2$~\cite[p.~220]{Peskin}, note that
	\eqn{dDeldp}{
		\dv{\Del}{\psl} = \dv{\psl}[ -x (1 - x) \psl^2 + x \mu^2 + (1 - x) \mo^2 ]
		= -2x (1 - x) \psl.
	}
	Also, ignoring terms of $\order{\Lam^{-2}}$,
	\aln{ \label{others}
		\dv{\psl}(-2 x \psl + 4 \mo) &= -2x, \\
		\dv{\Del}\brac{ \ln(\frac{\Del + \Lam^2}{\Del}) - 1 } &= \dv{\Del}\brac{ \ln(\Del + \Lam^2) - \ln(\Del) - 1 }
		= \frac{1}{\Del + \Lam^2} - \frac{1}{\Del}
		\approx -\frac{1}{\Del}.
	}
	Making these substitutions in Eq.~\refeq{der},
	\eq{
		\dv{\Sigw}{\psl} = \frac{e^2}{16 \pi^2} \intoq \ddx \brac{ -2x \brac{ \ln(\frac{\Del + \Lam^2}{\Del}) - 1 } - \frac{(2 x \psl - 4 \mo) [ 2x (1 - x) \psl ]}{\Del} }.
	}
	We now define
	\eqn{Delm}{
		\Delm \equiv -x (1 - x) m^2 + x \mu^2 + (1 - x) \mo^2
		\approx (1 - x)^2 m^2 + x \mu^2,
	}
	since $m \approx \mo$.  Then Eq.~\refeq{Z2} becomes
	\aln{
		\del\Zw &= \frac{e^2}{16 \pi^2} \intoq \ddx \brac{ -x \brac{ \ln(\frac{\Delm + \Lam^2}{\Delm}) - 1 } + \frac{(2 x m - 4 m) [ 2x (1 - x) m ]}{\Delm} } \notag \\
		&= -\frac{e^2}{8 \pi^2} \intoq \ddx \brac{ x \ln(\frac{\Delm + \Lam^2}{\Delm}) - x - \frac{2x m^2 (2 - x) (1 - x)}{\Delm} } \label{Z2ans}
	}
	Now we rename $x \to z$ in $\del\Zw$.  This means $\Delo = \Delm$ from Eqs.~\refeq{Delo} and \refeq{Delm}.  Naming $\Del \equiv \Delo = \Delm$ in Eqs.~\refeq{Z1ans} and \refeq{Z2ans}, we have
	\ans{ \al{
		\del\Zq &= \frac{e^2}{8 \pi^2} \intoq \ddz (1 - z) \brac{ \ln(\frac{\Del +\Lam^2}{\Del}) - \frac{3}{2} - \frac{m^2 (1 - 4z + z^2)}{\Del} }, \\
		\del\Zw &= -\frac{e^2}{8 \pi^2} \intoq \ddx \brac{ x \ln(\frac{\Del + \Lam^2}{\Del}) - x - \frac{2x m^2 (2 - x) (1 - x)}{\Del} },
	}}%
	where $\Del = (1 - z)^2 m^2 + z \mu^2$.  We see that \ans{$\del\Zq \neq \del\Zw$}, as we wanted to show. \qed
	
	\hl{fix log arguments and integrate}
}

\prob{ \label{2a}

\sol{
	We begin by finding $\del\Zq$.  We are interested in the diagram 
	\eqn{diag}{
		\centergraphics{diag/loop}
	}
	We considered a similar diagram in Homework~6 of Physics~443, and we build on that work here.  We adapted Peskin \& Schroeder (6.38) using the scalar field Feynman rules to write~\cite[p.~123]{Peskin}
	\aln{
		\ubpp \del\Gamm(p, p') \up &= \int \dddkf \frac{i}{(k - p)^2 - \mphi^2 + i \eps} \ubpp \paren{ -i \frac{\lam}{\sqrt{2}} } \frac{i (\ksl' + \me)}{{k'}^2 - \me^2 + i \eps} \gamm \frac{i (\ksl + \me)}{k^2 - \me^2 + i \eps} \paren{ -i \frac{\lam}{\sqrt{2}} } \up \notag \\

	}
	We use Peskin \& Schroeder~(6.41) to write 
	\eqn{denom}{
		\frac{1}{[ (k - p)^2 - \mphi^2 + i \eps ] ({k'}^2 - \me^2 + i \eps) (k^2 - \me^2 + i \eps)} = \intoq \ddx \ddy \ddz \del(x + y + z - 1) \frac{2}{D^3},
	}
	where~\cite[pp.~190--191]{Peskin}
	\eqn{D}{
		D = k^2 + 2 k (q y - p z) + z (p^2 - \mphi^2) - (1 - z) \me^2 + i \eps
		\equiv \ell^2 - \Del + i \eps,
	}
	where we use $x + y + z = 1$ and $k' = k + q$, and we have defined $\ell \equiv k + y q - z p$~\cite[p.~191]{Peskin}, and
	\eq{
		\Del \equiv -x y q^2 + (1 - z)^2 \me^2 + z \mphi^2.
	}
	For the numerator of Eq.~\refeq{thing1a}, we use $k' = k + q$ and $\ell \equiv k + y q - z p$~\cite[p.~191]{Peskin}, and define
	\eqn{N1}{
		N \equiv \ubpp [ \lsl + (1 - y) \qsl + z \psl + \me ] \gamm (\lsl - y \qsl + z \psl + \me) \up
	}
	To simplify $N$, we can discard terms that are linear in $\ell$ by~(6.45),
	\eq{
		\int \dddlf \frac{\ellm}{D^3} = 0.
	}
	Then Eq.~\refeq{N1} simplifies to
	\aln{
		N &= \ubpp [ \lsl \gamm \lsl - y (1 - y) \qsl \gamm \qsl + z (1 - y) \qsl \gamm \psl + \me (1 - y) \qsl \gamm - y z \psl \gamm \qsl + z^2 \psl \gamm \psl + \me z \psl \gamm - \me y \gamm \qsl \notag \\
		&\hspace{5em} \phantom{=\ } + \me z \gamm \psl + \me^2 \gamm ] \up \label{N2}
	}
	To simplify each of these terms we apply~(7.87),
	\eq{
		\int \dddlf \frac{\ellm \elln}{D^3} = \int \dddlf \frac{\gmn \ell^2}{g D^3},
	}
	as well as~\cite[pp.~191--192]{Peskin}
	\al{
		\psl \gamm &= 2 \pmu - \gamm \psl, &
		\psl \up &= \me \up, &
		\ubpp \psl' &= \ubpp \me,
	}
	and the Gordon identity
	\eq{
		\ubpp \gamm \up = \ubpp \brac{ \frac{{\pmu}' + \pmu}{2 m} + \frac{i \sigmn \qsn}{2m} } \up.
	}
	From Eq.~\refeq{7.47}, we can set $q = 0$.  Since $q = p' - p$, this means $p' = p$, and the Gordon identity becomes
	\eq{
		\ubpp \gamm \up = \ubpp \frac{\pmu}{m} \up
		\qimplies \pmu \to \me \gamm.
	}
	So the terms in Eq.~\refeq{N2} reduce to
	\al{
		\lsl \gamm \lsl &= (2 \ellm - \gamm \lsl) \lsl
		= 2 \ellm \elln \gamsn - \gamm \lsl \lsl
		\to \frac{2}{d} \ell^2 \gmn \gamsn - \gamm \ell^2
		\to \frac{2 - d}{d} \ell^2 \gamm - 2 \me^2 \gamm
		= 0,
		\\
		\qsl \gamm \qsl &\to  0,
		\\
		\qsl &\to 0,
		\\
		\qsl \gamm &\to 0,
		\\
		\psl \gamm \qsl &\to 0, \\
		\psl \gamm \psl &\to \me \psl \gamm
		= \me (2 \pmu - \gamm \psl)
		\to 2 \me \pmu - \me^2 \gamm
		\to 2 \me^2 \gamm - \me^2 \gamm
		= \me^2 \gamm,
		\\
		\psl \gamm &= 2 \pmu - \gamm \psl
		= 2 \pmu - \gamm \me
		\to 2 \me \gamm - \gamm \me
		= \me \gamm,
		\\
		\gamm \qsl &\to 0,
		\\
		\gamm \psl &\to \me \gamm.
	}
	Feeding these into Eq.~\refeq{N2}, we obtain
	\al{
		N \to \ubpp \brac{ \frac{2 - d}{d} \ell^2 + z^2 \me^2 + \me^2 z + \me^2 z + \me^2 } \gamm \up
		= \ubpp \brac{ \frac{2 - d}{d} \ell^2 + (1 + z)^2 \me^2 } \gamm \up.
	}
	Then, letting $\Delo = (1 - z)^2 \me^2 + z \mphi^2$,
	\aln{
		\del\Zq &= -\del\Fq \notag \\
		&= -i \lam^2 \int \dddlf \intoq \ddx \ddy \ddz \del(x + y + z - 1) \frac{(2 - d) \ell^2 / d + (1 + z)^2 \me^2}{(\ell^2 - \Delo)^3}  \notag \\
		&= -i \lam^2 \int \dddlf \intoq \ddz (1 - z) \frac{(2 - d) \ell^2 / d + (1 + z)^2 \me^2}{(\ell^2 - \Delo)^3}.  \label{thing3a}
	}
	Using (7.85) and (7.86) and substituting $\eps = 4 - d$, we can write
	\aln{
		\int \dddlf \frac{1}{(\ell^2 - \Del)^3} &= \frac{i}{(-1)^3} \frac{1}{(2\pi)^d} \int \frac{\dddlE}{(\ellE^2 + \Del)^3} \notag \\
		&= -i \int \dddlEf \frac{1}{(\ellE^2 + \Del)^3} \notag \\
		&= -\frac{i}{(4\pi)^{d / 2}} \frac{\Gam(3 - d/2)}{\Gam(3)} \paren{ \frac{1}{\Del} }^{3 - d/2} \notag \\
		&= -\frac{i}{(4\pi)^{2 - \eps / 2}} \frac{\Gam(1 + \eps / 2)}{\Gam(3)} \paren{ \frac{1}{\Del} }^{1 + \eps / 2} \label{int1}
		\\[2ex]
		\int \dddlf \frac{\ell^2}{(\ell^2 - \Del)^3} &= -\frac{i}{(-1)^3} \frac{1}{(2\pi)^d} \int \dddlE \frac{\ellE^2}{(\ellE^2 + \Del)^3} \notag \\
		&= i \int \dddlEf \frac{1}{(\ellE^2 + \Del)^3} \notag \\
		&= \frac{i}{(4\pi)^{d / 2}} \frac{d}{2} \frac{\Gam(2 - d/2)}{\Gam(3)} \paren{ \frac{1}{\Del} }^{2 - d/2} \notag \\
		&= \frac{i}{(4\pi)^{2 - \eps / 2}} \frac{4 - \eps}{2} \frac{\Gam(\eps / 2)}{\Gam(3)} \paren{ \frac{1}{\Del} }^{\eps / 2}. \label{int2}
	}
	Then Eq.~\refeq{thing3a} becomes
	\aln{
		\del\Zq &= -i \lam^2 \intoq \ddz (1 - z) \brac{ \frac{2 - d}{d} \frac{i}{(4\pi)^{2 - \eps / 2}} \frac{d}{2} \frac{\Gam(\eps / 2)}{\Gam(3)} \paren{ \frac{1}{\Delo} }^{\eps / 2} - (1 + z)^2 \me^2 \frac{i}{(4\pi)^{2 - \eps / 2}} \frac{\Gam(1 + \eps / 2)}{\Gam(3)} \paren{ \frac{1}{\Delo} }^{1 + \eps / 2} } \notag \\
		&= \frac{1}{\Gam(3)} \frac{\lam^2}{(4\pi)^{2 - \eps / 2}} \intoq \ddz (1 - z) \brac{ \frac{\eps - 2}{2} \Gam(\eps / 2) \paren{ \frac{1}{\Delo} }^{\eps / 2} - (1 + z)^2 \me^2 \Gam(1 + \eps / 2) \paren{ \frac{1}{\Delo} }^{1 + \eps / 2} } \notag \\
		&= \frac{1}{\Gam(3)} \frac{\lam^2}{(4\pi)^2} \intoq \ddz (1 - z) \brac{ \frac{\eps - 2}{2} \Gam(\eps / 2) \paren{ \frac{4\pi}{\Delo} }^{\eps / 2} - \frac{(1 + z)^2 \me^2}{\Del} \Gam(1 + \eps / 2) \paren{ \frac{4\pi}{\Delo} }^{\eps / 2} } \notag \\
		&\approx -\frac{\lam^2}{32 \pi^2} \intoq \ddz (1 - z) \brac{ \frac{2}{\eps} - \gam + \ln(\frac{4\pi}{\Del}) + \frac{(1 + z)^2 \me^2}{\Del} }, \label{Z1a}
	}
	where we have performed the Taylor series expansion about $\eps = 0$ using Mathematica, keeping terms larger than $\order{\eps}$.
	
	Now we must find $\del\Zw$.  We begin with a diagram that looks like that in (7.15) of Peskin \& Schroeder, but with a $\phi$ instead of a photon.  We need to adapt their (7.16) for a scalar field.  Using the Feynman rules for a scalar field~\cite[p.~118]{Peskin}, we write
	\eq{
		-i \Sigwp = \paren{ -\frac{i \lam}{\sqrt{2}} }^2 \int \dddkf \frac{i (\ksl + \me)}{k^2 - \me^2 + i \eps} \frac{i}{(p - k)^2 - \mphi^2 + i \eps}.
	}
	Let $\ell = k - x p$ and $\Del = -x (1 - x) p^2 + x \mphi^2 + (1 - x) \me^2$.  Then from Eq.~\refeq{Sigw2},
	\eqn{Siqwa.1}{
		\Sigwp = \frac{i \lam^2}{2} \intoq \ddx \int \dddlf \frac{\lsl + x \psl + \me}{(\ell^2 - \Del + i \eps)^2}
		\to \frac{i \lam^2}{2} \intoq \ddx \int \dddlf \frac{x \psl + \me}{(\ell^2 - \Del + i \eps)^2},
	}
	where we have again dropped terms linear in $\ell$.  Then from (7.85),
	\al{
		\int \dddlf \frac{1}{(\ell^2 - \Del)^2} &= \frac{i}{(-1)^2} \frac{1}{(2\pi)^d} \int \frac{\dddlE}{(\ellE^2 + \Del)^2} \\
		&= i \int \dddlEf \frac{1}{(\ellE^2 + \Del)^2} \\
		&= \frac{i}{(4\pi)^{d / 2}} \frac{\Gam(2 - d/2)}{\Gam(2)} \paren{ \frac{1}{\Del} }^{2 - d/2} \\
		&= \frac{i}{(4\pi)^{2 - \eps / 2}} \frac{\Gam(\eps / 2)}{\Gam(2)} \paren{ \frac{1}{\Del} }^{\eps / 2}.
	}
	Feeding this into Eq.~\refeq{Sigwa1} and Taylor expanding as before,
	\al{
		\Sigwp &= \frac{i \lam^2}{2} \intoq \ddx (x \psl + \me) \frac{i}{(4\pi)^{2 - \eps / 2}} \frac{\Gam(\eps / 2)}{\Gam(2)} \paren{ \frac{1}{\Del} }^{\eps / 2} \\
		&= -\frac{\lam^2}{32 \pi^2} \intoq \ddx (x \psl + \me) \Gam(\eps / 2) \paren{ \frac{4\pi}{\Del} }^{\eps / 2} \\
		&\approx -\frac{\lam^2}{32 \pi^2} \intoq \ddx (x \psl + \me) \brac{ \frac{2}{\eps} - \gam + \ln(\frac{4\pi}{\Del}) }.
	}
	Note that
	\eq{
		\dv{\Sigwp}{\psl} = -\frac{\lam^2}{32 \pi^2} \intoq \ddx \curly{ \brac{ \frac{2}{\eps} - \gam + \ln(\frac{4\pi}{\Del}) } \dv{\psl}(x \psl + \me) + (x \psl + \me) \dv{\Del}\brac{ \frac{2}{\eps} - \gam + \ln(\frac{4\pi}{\Del}) } \dv{\Del}{\psl} },
	}
	where
	\al{
		\dv{\psl}(x \psl + \me) &= x, &
		\dv{\Del}\brac{ \frac{2}{\eps} - \gam + \ln(\frac{4\pi}{\Del}) } &= -\frac{1}{\Del}, &
		\dv{\Del}{\psl} &= -2x (1 - x) \psl.
	}
	Then
	\eq{
		\dv{\Sigwp}{\psl} = -\frac{\lam^2}{32 \pi^2} \intoq \ddx \curly{ x \brac{ \frac{2}{\eps} - \gam + \ln(\frac{4\pi}{\Del}) } + (x \psl + \me) \frac{2x (1 - x) \psl}{\Del} },
	}
	so from Eq.~\refeq{Z2},
	\eq{
		\del\Zw = -\frac{\lam^2}{32 \pi^2} \intoq \ddx x \brac{ \frac{2}{\eps} - \gam + \ln(\frac{4\pi}{\Delm}) + \frac{2 \me^2 (1 + x) (1 - x)}{\Delm} },
	}
	where $\Delm = (1 - x)^2 \me^2 + x \mphi^2$.
	
	\hl{DON'T FORGET TO DO THE MATH}
}



\prob{
	Now consider the renormalization of the $\phi \psib \psi$ vertex.  Show that the rescaling of this vertex at $q^2 = 0$ is \emph{not} canceled by the correction to $\Zw$.  (It suffices to compute the ultraviolet-divergent parts of the diagrams.)  In this theory, the vertex and field-strength rescaling give additional shifts of the observable coupling constant relative to its bare value.
}

\sol{
	In order to account for both terms of the Hamiltonian, we need to sum the diagrams with scalar propagators and with photon propagators.  Both of the diagrams have scalar external legs.  This means we need to replace Eq.~\refeq{Z1} with a relation that makes sense for a scalar vertex.  We adapt (7.47) as
	\eqn{Gamscalar}{
		\Gam(q = 0) = \Zq^{-1}
		\qimplies
		\del\Gam(q = 0) = -\del\Zq.
	}
	To compute $\del\Zq$, we must sum two vertex diagrams similar to the one on p.~189 of Peskin \& Schroeder.  The first has the external leg replaced with a scalar:
	\aln{
		\ubpp \del\Gamq(p, p') \up &= \int \dddkf \frac{-i \gsnr}{(k - p)^2 + i \eps} \ubpp (-i e \gamn) \frac{i (\ksl' + \me)}{{k'}^2 - \me^2 + i \eps} \frac{i (\ksl + \me)}{k^2 - \me^2 + i \eps} (-i e \gamr) \up \notag \\
		&= -i e^2 \int \dddkf \ubpp \gamn \frac{(\ksl' + \me) (\ksl + \me)}{[ (k - p)^2 + i \eps ] [ {k'}^2 - \me^2 + i \eps ] [ k^2 - \me^2 + i \eps ]} \gamsn \up \label{Gam1}
	}
	We can simplify the calculations at this step (contrary to what we did in \ref{2a}) by setting $q = k' - k = 0$ now.  We introduce Feynman parameters analogously to Eq.~\refeq{denom}.  Let $\ell = k - x p$.  We also introduce the photon mass now, and so define $\Delq = (1 - z)^2 m^2 + z \mu^2$ as in Eq.~\refeq{Delo}.  Then the denominator of the integrand is $\Dq = \ell^2 + \Delq + i \eps$, and the numerator is
	\al{
		\Nq &= \ubpp \gamn (\ksl + \me) (\ksl + \me) \gamsn \up \\
		&= \ubpp \gamn (\lsl + z \psl + \me) (\lsl + z \psl + \me) \gamsn \up \\
		&= \ubpp \gamn (\lsl \lsl + z \lsl \psl + \me \lsl + z^2 \psl \lsl + z \psl \psl + z \me \psl + \me \lsl + z \me \psl + \me^2) \gamsn \up \\
		&\to \ubpp \gamn (\ell^2 + z^2 \me^2 + 2 z \me \psl + \me^2) \gamsn \up \\
		&= \ubpp (d \ell^2 + d z^2 \me^2 + 2 z \me \gamn \psl \gamsn + d \me^2) \up \\
		&= \ubpp [ d \ell^2 + d z^2 \me^2 + 2 z \me \gamn (2 \psn - \gamsn \psl) + d \me^2 ] \up \\
		&= \ubpp [ d \ell^2 + d z^2 \me^2 + 2 z \me (2 \psl - d \psl) + d \me^2 ] \up \\
		&= \ubpp [ d \ell^2 + d \me^2 (z^2 + 1) + 2 z \me^2 (2 - d) ] \up,
	}
	where we have dropped terms linear in $\ell$.  Then applying Eqs.~\refeq{Gamscalar} and \refeq{Gam1}, we can write an expression for the contribution to $\del\Zq$, which we call $\del\Zqq$.  We also make use of Eqs.~\refeq{int1} and \refeq{int2}.  We find
	\aln{
		\del \Zqq &= i e^2\intoq \ddz (1 - z) \int \dddlf \frac{d \ell^2 + d \me^2 (z^2 + 1) + 2 z \me^2 (2 - d)}{(\ell^2 + \Delq + i \eps)^3} \notag \\[2ex]
		%
		&= i e^2 \intoq \ddz (1 - z) \left[ \frac{i}{(4\pi)^{2 - \eps / 2}} \frac{(4 - \eps)^2}{2} \frac{\Gam(\eps / 2)}{\Gam(3)} \paren{ \frac{1}{\Delq} }^{\eps / 2} \right. \notag \\
		&\hspace{10em} \phantom{=\ } \left. + \me^2 [ (4 - \eps) (z^2 + 1) + 2 z (\eps - 2) ] \paren{ -\frac{i}{(4\pi)^{2 - \eps / 2}} \frac{\Gam(1 + \eps / 2)}{\Gam(3)} \paren{ \frac{1}{\Delq} }^{1 + \eps / 2} } \notag \right] \notag \\[2ex]
		%
		&= -\frac{e^2}{32 \pi^2} \intoq \ddz (1 - z) \left[ \frac{(4 - \eps)^2}{2} \Gam(\eps / 2) \paren{ \frac{4\pi}{\Delq} }^{\eps / 2} - \frac{\me^2 (z^2 + 1)}{\Delq} (4 - \eps) \Gam(1 + \eps / 2) \paren{ \frac{4\pi}{\Delq} }^{\eps / 2} \right. \notag \\
		&\hspace{25em} \phantom{=\ } \left. - \frac{2 \me^2 z}{\Delq} (\eps - 2) \Gam(1 + \eps / 2) \paren{ \frac{4\pi}{\Delq} }^{\eps / 2} \right] \notag \\[2ex]
		%
		&\approx -\frac{e^2}{32 \pi^2} \intoq \ddz (1 - z) \brac{ \frac{16}{\eps} - 8 - 8 \gam + 8 \ln(\frac{4\pi}{\Delq}) - 4 \frac{\me^2 (z^2 + 1)}{\Delq} + 4 \frac{\me^2 z}{\Delq} } \notag \\[2ex]
		%
		&= -\frac{e^2}{8 \pi^2} \intoq \ddz (1 - z) \brac{ \frac{4}{\eps} - 2 - 2 \gam + 2 \ln(\frac{4\pi}{\Delq}) - \frac{\me^2 (z^2 - z + 1)}{\Delq} }. \label{delZqq}
	}
	
	The second Feynman diagram looks like Eq.~\refeq{diag}, but with a scalar loop instead of a photon loop:
	\aln{
		\ubpp \del\Gamw(p, p') \up &= \int \dddkf \frac{i}{(k - p)^2 - \mphi^2 + i \eps} \ubpp \paren{ -\frac{i \lam}{\sqrt{2}} } \frac{i (\ksl' + \me)}{{k'}^2 - \me^2 + i \eps} \frac{i (\ksl + \me)}{k^2 - \me^2 + i \eps} \paren{ -\frac{i \lam}{\sqrt{2}} } \up \notag \\
		&= \frac{i \lam^2}{2} \int \dddkf \ubpp \frac{(\ksl' + \me) (\ksl + \me)}{[ (k - p)^2 - \mphi^2 + i \eps ] [ {k'}^2 - \me^2 + i \eps ] [ k^2 - \me^2 + i \eps ]} \up \label{Gam2}
	}
	Again setting $q = 0$, we again apply the appropriate analogue to Eq.~\refeq{denom} and let $\Delw = (1 - z)^2 \me^2 + z \mphi^2$.  Then the denominator of the integrand is $\Dw = \ell^2 - \Delw + i \eps$, and the numerator is (again dropping terms linear in $\ell$)
	\al{
		\Nw &= \ubpp (\ksl + \me) (\ksl + \me) \ub \\
		&= \ubpp (\lsl \lsl + x \lsl \psl + \me \lsl + x \psl \lsl + x \psl \psl + x \me \psl + \me \lsl + x \me \psl + \me^2) \ub \\
		&= \ubpp (\ell^2 + 2 x \lsl \me + 2 \me \lsl + x^2 \me^2 + 2 x \me^2 + \me^2) \ub \\
		&\to \ubpp [ \ell^2 + \me^2 (1 + x)^2 ] \ub.
	}
	We call the contribution to $\del\Zq$ from this diagram $\del\Zqw$.  Again using Eqs.~\refeq{int1} and \refeq{int2} yields
	\al{
		\del\Zqw &= -\frac{i \lam^2}{2} \intoq \ddz (1 - z) \int \dddkf \frac{\ell^2 + \me^2 (1 + x)^2}{(\ell^2 - \Delw + i \eps)^3} \\
		&= -\frac{i \lam^2}{2} \intoq \ddz (1 - z) \brac{ \frac{i}{(4\pi)^{2 - \eps / 2}} \frac{4 - \eps}{2} \frac{\Gam(\eps / 2)}{\Gam(3)} \paren{ \frac{1}{\Delw} }^{\eps / 2} - \me^2 (1 + x)^2 \frac{i}{(4\pi)^{2 - \eps / 2}} \frac{\Gam(1 + \eps / 2)}{\Gam(3)} \paren{ \frac{1}{\Delw} }^{1 + \eps / 2} } \\
		&= \frac{\lam^2}{64 \pi^2} \intoq \ddz (1 - z) \brac{ \frac{4 - \eps}{2} \Gam(\eps / 2) \paren{ \frac{4\pi}{\Delw} }^{\eps / 2} - \frac{\me^2 (1 + x)^2}{\Delw} \Gam(1 + \eps / 2) \paren{ \frac{4\pi}{\Delw} }^{1 + \eps / 2} } \\
		&\approx \frac{\lam^2}{64 \pi^2} \intoq \ddz (1 - z) \brac{ \frac{4}{\eps} - 1 - 2 \gam + 2 \ln(\frac{4 \pi}{\Delw}) - \frac{\me^2 (1 + x)^2}{\Delw} }.
	}
	With Eq.~\refeq{delZqq}, we find
	\al{
		\del\Zq &= \intoq \ddz (1 - z) \left\{ \frac{\lam^2}{64 \pi^2} \brac{ \frac{4}{\eps} - 1 - 2 \gam + 2 \ln(\frac{4 \pi}{\Delw}) - \frac{\me^2 (1 + x)^2}{\Delw} } \right. \\
		&\hspace{15em} \phantom{=\ } \left. - \frac{e^2}{8 \pi^2} \brac{ \frac{4}{\eps} - 2 - 2 \gam + 2 \ln(\frac{4\pi}{\Delq}) - \frac{\me^2 (z^2 - z + 1)}{\Delq} } \right\}.
	}
}
