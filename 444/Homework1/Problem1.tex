\state{Alternative regulators in QED (Peskin \& Schroeder 7.2)}{
	In Section.~7.5, we saw that the Ward identity can be violated by an improperly chosen regulator.  Let us check the validity of the identity $\Zq = \Zw$, to order $\alp$, for several choices of the regulator.  We have already verified that the relation holds for Paul-Villars regularization.
}

\prob{
	Recompute $\del\Zq$ and $\del\Zw$, defining the integrals~(6.49) and (6.50) by simply placing an upper limit $\Lam$ on the integration over $\ellE$.  Show that, with this definition, $\del\Zq \neq \del\Zw$.
}

\sol{
	From (7.47) in Peskin \& Schroeder,
	\eqn{7.47}{
		\Gammqo = \frac{1}{\Zq} \gamm,
	}
	we can find an expression for $\del\Zq$, similar to how (7.31) is obtained from (7.26).  Roughly,
	\eqn{thingy}{
		\frac{1}{\Zq + \del\Zq} \gamm \approx \Zq (1 - \del\Zq) \gamm
		= \Gammqo + \del\Gammqo
		\qimplies
		\del\Gammqo = -\del\Zq \gamm.
	}
	According to (6.33),
	\eq{
		\Gammppp = \gamm \Fq(q^2) + \frac{i \sigmn \qsn}{2 m} \Fw(q^2).
	}
	We note that $\Gamm = \gamm$, $\Fq = 1$, and $\Fw = 0$ to lowest order~\cite[pp.~185--186]{Peskin}.  Then we can write
	\eqn{Gam}{
		\del\Gammqo = \gamm \del\Fqo + \frac{i \sigmn \qsn}{2 m} \del\Fwo.
	}
	Using this equation and the identity $\gamm \gamsm = 4$~\cite{Gamma}, Eq.~\refeq{thingy} can be written
	\eqn{Z1}{
		\del \Zq = -\frac{1}{4} \gamsm \del\Gammqo
		= -\del\Fqo - \gamsm \frac{i \sigmn \qsn}{8 m} \del\Fwo.
	}
	
	In order to find $\del\Gamm$ we use (6.47):
	\aln{
		\ubpp \del\Gammppp \up &= 2 i e^2 \int \ddqlf \intoq \ddx \ddy \ddz \del(x + y + z - 1) \frac{2}{D^3} \notag \\
		&\hspace{5em} \phantom{=\ } \times \ubpp \left\{ \gamm \brac{ -\frac{\ell^2}{2} + (1 - x) (1 - y) q^2 + (1 - 4z + z^2) m^2 } \right. \label{thing1} \\
		&\hspace{15em} \phantom{=\ } \left. + \frac{i \sigmn \qsn}{2 m} [ 2 m^2 z (1 - z) ] \right\} \up, \notag
	}
	where $\Del \equiv -x y q^2 + (1 - z)^2 m^2$ by (6.44), $\ell \equiv k + y q - z p$, and $D = \ell^2 - \Del + i \eps$~\cite[p.~191]{Peskin}.  The momenta $k$ and $p$ are assigned in the Feynman diagram on p.~189 of Peskin \& Schroeder, and $x, y$ are Feynman parameters~\cite[p.~190]{Peskin}.

	The forms of the two integrals we need to compute are given by Peskin \& Schroeder~(6.49) and (6.50).  Equation~(6.49) is
	\eqn{6.49}{
		\int \ddqlf \frac{1}{(\ell^2 - \Delta)^m} = \frac{i (-1)^m}{(4\pi)^2} \frac{1}{(m - 1) (m - 2)} \frac{1}{\Delta^{m - 2}}.
	}
	Here $m = 3$ because we have $D^{-3}$ in Eq.~\refeq{thing1}.  We can evaluate the left-hand side using the Euclidian 4-momentum defined in (6.48),
	\aln{ \label{ellE}
		\ello &\equiv i \ellEo, &
		\vell &= \vellE.
	}
	Following the steps on p.~193, we have
	\eq{
		\int \ddqlf \frac{1}{(\ell^2 - \Delta)^3} = \frac{i}{(-1)^3} \frac{1}{(2\pi)^4} \int \frac{\ddqlE}{(\ellE^2 + \Del)^3}
		= \frac{i (-1)^3}{(2\pi)^4} \int \ddOmgq \intoLam \ddlE \frac{\ellE^3}{(\ellE^2 + \Delta)^3},
	}
	where we have replaced the upper (infinite) bound of integration by a finite number $\Lam$.  Evaluating this integral using Mathematica and using $\int \ddOmgq = 2 \pi^2$~\cite[p.~193]{Peskin}, we find
	\aln{
		\int \ddqlf \frac{1}{(\ell^2 - \Del)^3} &= \frac{i (-1)^3}{(2\pi)^4} (2 \pi^2) \frac{\Lam^4}{4 \Del (\Del + \Lam^2)^2} \notag \\
		&= -\frac{i}{32 \pi^2} \frac{\Lam^4}{\Del (\Del + \Lam^2)^2} \notag \\
		&\approx -\frac{i}{32 \pi^2} \frac{1}{\Del}
		\equiv \alp, \label{ans1}
	}
	where we have taken the limit $\Lam \gg \Del$~\cite[p.~218]{Peskin} and defined $\alp$.  Equation~(6.50) in Peskin \& Schroeder is
	\eq{
		\int \ddqlf \frac{\ell^2}{(\ell^2 - \Del)^m} = \frac{i (-1)^{m - 1}}{(4\pi)^2} \frac{2}{(m - 1) (m - 2) (m - 3)} \frac{1}{\Del^{m - 3}}.
	}
	Following similar steps as for Eq.~\refeq{thing1}, the left-hand side is
	\aln{
		\int \ddqlf \frac{\ell^2}{(\ell^2 - \Del)^3} &= \frac{i}{(-1)^3} \frac{1}{(2\pi)^4} \int \ddqlE \frac{\ellE^2}{(\ellE^2 + \Del)^3} \notag \\
		&= \frac{i (-1)^3}{(2\pi)^4} \int \ddOmgq \intoLam \ddlE \frac{\ellE^5}{(\ellE^2 + \Del)^3} \notag \\
		&= \frac{i (-1)^3}{(2\pi)^4} (2 \pi^2) \frac{1}{4} \brac{ \frac{\Del (3 \Del + 4 \Lam^2)}{(\Del + \Lam^2)^2} + 2 \ln(\frac{\Del + \Lam^2}{\Del}) - 3 } \notag \\
		&= -\frac{i}{32 \pi^2} \brac{ \frac{\Del (3 \Del + 4 \Lam^2)}{(\Del + \Lam^2)^2} + 2 \ln(\frac{\Del + \Lam^2}{\Del}) - 3 } \notag \\
		&\approx -\frac{i}{16 \pi^2} \brac{ \ln(\frac{\Lam^2}{\Del}) - \frac{3}{2} }
		\equiv \bet, \label{ans2}
	}
	where we have defined $\bet$ and ignored terms of $\order{\Lam^{-2}}$~\cite[p.~218]{Peskin}.
	
	We now set $q^2 = 0$, and define $\Delo = (1 - z)^2 m^2$.  Then $\Del \to \Delo$ in our expression and $\alp \to \alpo, \bet \to \beto$ (which are functions of $\Delo$).  Feeding in Eqs.~\refeq{ans1} and \refeq{ans2}, Eq.~\refeq{thing1} can be written
	\eq{
		\ubpp \del\Gammqo \up = 2 i e^2 \intoq \ddx \ddy \ddz \del(x + y + z - 1) \ubpp \int \curly{ \gamm \brac{ -\beto + 2 (1 - 4z + z^2) m^2 \alpo } } \up.
	}
	Then
	\al{
		\del\Fqo &= 2 i e^2 \intoq \ddx \ddy \ddz \del(x + y + z - 1) \brac{ -\beto + 2 (1 - 4z + z^2) m^2 \alpo } \\
		&= 2 i e^2 \intoq \ddz (1 - z) \brac{ -\beto + 2 m^2 (1 - 4z + z^2) \alpo }, \\[2ex]
		%
		\del\Fwo &= 8 i e^2 \intoq \ddx \ddy \ddz \del(x + y + z - 1) m^2 z (1 - z) \alpo \\
		&= 8 i e^2 \intoq \ddz m^2 z (1 - z)^2 \alpo.
	}
	We ignore $\del\Fwo$ since it is not affected by the divergence~\cite[p.~196]{Peskin}.  In order to avoid issues coming from the divergence in $\del\Fqo$, we add a $z \mu^2$ term to $\Delo$~\cite[p.~195]{Peskin}.  So, feeding these results into Eq.~\refeq{Z1}, we obtain
	\aln{
		\del\Zq &= -2 i e^2 \intoq \ddz (1 - z) \brac{ -\beto + 2 (1 - 4z + z^2) m^2 \alpo } \notag \\
		&= -2 i e^2 \intoq \ddz (1 - z) \curly{ \frac{i}{16 \pi^2} \brac{ \ln(\frac{\Lam^2}{\Delo}) - \frac{3}{2} } - 2 (1 - 4z + z^2) m^2 \frac{i}{32 \pi^2} \frac{1}{\Delo} } \notag \\
		&= \frac{e^2}{8 \pi^2} \intoq \ddz (1 - z) \brac{ \ln(\frac{\Lam^2}{\Delo}) - \frac{3}{2} - \frac{m^2 (1 - 4z + z^2)}{\Delo} }, \label{Z1ans}
	}
	where
	\eqn{Delo}{
		\Delo = (1 - z)^2 m^2 + z \mu^2.
	}
	
	For $\del\Zw$, we can use the first part of Peskin \& Schroeder (7.31),
	\eqn{Z2}{
		\del\Zw = \left. \dv{\Sigw}{\psl} \right|_{\psl = m},
	}
	where $\Sigw$ is given by (7.17),
	\eqn{Sigw}{
		-i \Sigwp = -e^2 \intoq \ddx \int \ddqlf \frac{-2 x \psl + 4 \mo}{(\ell^2 - \Del + i \eps)^2},
	}
	where $\Del = -x (1 - x) p^2 + x \mu^2 + (1 - x) \mo^2$.  We may once again follow the steps on p.~193 to evaluate the integral, now with $m = 2$.  Changing the upper bound of integration to $\Lam$ once more, we have
	\al{
		\int \ddqlf \frac{1}{(\ell^2 - \Delta)^2} &= \frac{i}{(-1)^2} \frac{1}{(2\pi)^4} \int \ddqlE \frac{1}{(\ellE^2 + \Del)^2} \\
		&= \frac{i (-1)^2}{(2\pi)^4} \int \ddOmgq \intoLam \ddlE \frac{\ellE^3}{(\ellE^2 + \Del)^2} \\
		&= \frac{i}{(2\pi)^4} (2 \pi^2) \frac{1}{2} \brac{ \frac{\Del}{\Del + \Lam^2} + \ln(\frac{\Del + \Lam^2}{\Del}) - 1 } \\
		&= \frac{i}{16 \pi^2} \brac{ \frac{\Del}{\Del + \Lam^2} + \ln(\frac{\Del + \Lam^2}{\Del}) - 1 } \\
		&\approx \frac{i}{16 \pi^2} \brac{ \ln(\frac{\Lam^2}{\Del}) - 1 },
	}
	where we have evaluated the integral using Mathematica and dropped terms of $\order{\Lam^{-2}}$.  Substituting back into Eq.~\refeq{Sigw}, we find
	\eq{
		\Sigwp = -i e^2 \intoq \ddx (-2 x \psl + 4 \mo) \frac{i}{16 \pi^2} \brac{ \ln(\frac{\Lam^2}{\Del}) - 1 }.
	}
	Note that
	\aln{
		\dv{\Sigw}{\psl} &= \frac{e^2}{16 \pi^2} \dv{\psl} \curly{ \intoq \ddx (-2 x \psl + 4 \mo) \brac{ \ln(\frac{\Lam^2}{\Del}) - 1 } } \notag \\
		&= \frac{e^2}{16 \pi^2} \intoq \ddx \curly{ \brac{ \ln(\frac{\Lam^2}{\Del}) - 1 } \dv{\psl}(-2 x \psl + 4 \mo) + (-2 x \psl + 4 \mo) \dv{\psl} \brac{ \ln(\frac{\Lam^2}{\Del}) - 1 } } \notag \\
		&= \frac{e^2}{16 \pi^2} \intoq \ddx \curly{ \brac{ \ln(\frac{\Lam^2}{\Del}) - 1 } \dv{\psl}(-2 x \psl + 4 \mo) + (-2 x \psl + 4 \mo) \dv{\Del} \brac{ \ln(\frac{\Lam^2}{\Del}) - 1 } \dv{\Del}{\psl} }. \label{der}
	}
	Using $p^2 = \psl^2$~\cite[p.~220]{Peskin}, note that
	\eqn{dDeldp}{
		\dv{\Del}{\psl} = \dv{\psl}[ -x (1 - x) \psl^2 + x \mu^2 + (1 - x) \mo^2 ]
		= -2x (1 - x) \psl.
	}
	Also, ignoring terms of $\order{\Lam^{-2}}$,
	\aln{ \label{others}
		\dv{\psl}(-2 x \psl + 4 \mo) &= -2x, \\
		\dv{\Del}\brac{ \ln(\frac{\Lam^2}{\Del}) - 1 } &= \dv{\Del}\brac{ \ln(\Lam^2) - \ln(\Del) - 1 }
		= -\frac{1}{\Del}.
	}
	Making these substitutions in Eq.~\refeq{der},
	\eq{
		\dv{\Sigw}{\psl} = \frac{e^2}{16 \pi^2} \intoq \ddx \brac{ -2x \brac{ \ln(\frac{\Lam^2}{\Del}) - 1 } - \frac{(2 x \psl - 4 \mo) [ 2x (1 - x) \psl ]}{\Del} }.
	}
	We now define
	\eqn{Delm}{
		\Delm \equiv -x (1 - x) m^2 + x \mu^2 + (1 - x) \mo^2
		\approx (1 - x)^2 m^2 + x \mu^2,
	}
	since $m \approx \mo$.  Then Eq.~\refeq{Z2} becomes
	\aln{
		\del\Zw &= \frac{e^2}{16 \pi^2} \intoq \ddx \brac{ -x \brac{ \ln(\frac{\Lam^2}{\Delm}) - 1 } + \frac{(2 x m - 4 m) [ 2x (1 - x) m ]}{\Delm} } \notag \\
		&= -\frac{e^2}{8 \pi^2} \intoq \ddx \brac{ x \ln(\frac{\Lam^2}{\Delm}) - x - \frac{2x m^2 (2 - x) (1 - x)}{\Delm} } \label{Z2ans}
	}
	Now we rename $x \to z$ in $\del\Zw$.  This means $\Delo = \Delm$ from Eqs.~\refeq{Delo} and \refeq{Delm}.  Naming $\Del \equiv \Delo = \Delm$ in Eqs.~\refeq{Z1ans} and \refeq{Z2ans}, we have
	\ans{ \al{
		\del\Zq &= \frac{e^2}{8 \pi^2} \intoq \ddz (1 - z) \brac{ \ln(\frac{\Lam^2}{\Del}) - \frac{3}{2} - \frac{m^2 (1 - 4z + z^2)}{\Del} }, \\
		\del\Zw &= -\frac{e^2}{8 \pi^2} \intoq \ddx \brac{ x \ln(\frac{\Lam^2}{\Del}) - x - \frac{2x m^2 (2 - x) (1 - x)}{\Del} },
	}}%
	where $\Del = (1 - z)^2 m^2 + z \mu^2$.  It appears that $\del\Zq \neq \del\Zw$, as we wanted to show, but I am not sure how to evaluate these integrals.
}



\prob{ \label{1b}
	Recompute $\del\Zq$ and $\del\Zw$, defining the integrals~(6.49) and (6.50) by dimensional regularization.  You may take the Dirac matrices to be $4 \times 4$ as usual, but note that, in $d$ dimensions,
	\eqn{gamd}{
		\gsmn \gamm \gamn = d.
	}
	Show that, with this definition, $\del\Zq = \del\Zw$.
}

\sol{
	We begin by finding $\del\Zw$.  We need to fix Peskin \& Schroeder~(7.17) so it has arbitrary $d$ instead of $d = 4$.  We begin from (7.16), changing $4 \to d$:
	\eq{
		-i \Sigwp = (-ie)^2 \int \dddkf \gamm \frac{i (\ksl + \mo)}{k^2 + \mo^2 + i \eps} \gamsm \frac{i}{(p - k)^2 - \mu^2 + i \eps}.
	}
	Following the procedure on pp.~217--218, we introduce the Feynman parameter $x$ to combine the denominators:
	\eqn{FP}{
		\frac{1}{k^2 - \mo^2 + i \eps} \frac{1}{(p - k)^2 - \mu^2 + i \eps} = \intoq \ddx \frac{1}{[ k^2 - 2 x k \cdot p + x p^2 - x \mu^2 - (1 - x) \mo^2 + i \eps ]^2}.
	}
	Let $\ell = k - x p$ and $\Del = -x(1 - x) p^2 + x \mu^2 + (1 - x) \mo^2$.  Then
	\aln{
		-i \Sigwp &= (-ie)^2 \intoq \ddx \int \dddlf \gamm \frac{i^2 (\ksl + \mo)}{[ \ell^2 - \Del + i \eps ]^2} \gamsm \notag \\
		&= -(-ie)^2 \intoq \ddx  \int \dddlf \gamm \frac{\lsl + x \psl + \mo}{[ \ell^2 - \Del + i \eps ]^2} \gamsm \notag \\
		&= -e^2 \intoq \ddx \int \dddlf \gamm \frac{2 \ellm - \gamsm \lsl + x (2 \psm - \gamsm \psl) + \mo \gamsm}{[ \ell^2 - \Del + i \eps ]^2} \notag \\
		&= -e^2 \intoq \ddx \int \dddlf \frac{(2 - d) x \psl + d \mo}{[ \ell^2 - \Del + i \eps ]^2}, \label{Sigw2}
	}
	where we have applied Eq.~\refeq{gamd} and $\psl \gamm = 2 \pmu - \gamm \psl$~\cite[p.~191]{Peskin}, and we have dropped terms linear in $\ell$ by~(6.45),
	\eq{
		\int \dddlf \frac{\ellm}{D^3} = 0.
	}
	To evaluate the integral, we can write it in terms of the Euclidean 4-momentum defined in Eq.~\refeq{ellE}, as on p.~193 in Peskin \& Schroeder:
	\eq{
		\int \dddlf \frac{1}{(\ell^2 - \Del)^2} = \frac{i}{(-1)^2} \frac{1}{(2\pi)^d} \int \dddlE \frac{1}{(\ellE^2 + \Del)^2}
		= i \int \dddlEf \frac{1}{(\ellE^2 + \Del)^2}.
	}
	Then we can apply (7.84), which takes the limit as $d \to 4$:
	\eq{
		\int \dddlEf \frac{1}{(\ellE^2 + \Del)^2} \to \frac{1}{(4\pi)^2} \paren{ \frac{2}{\eps} - \gam + \ln(\frac{4\pi}{\Del}) },
	}
	where $\eps = 4 - d$~\cite[p.~250]{Peskin}.  Making these substitutions into Eq.~\refeq{Sigw2}, we find
	\eq{
		\Sigwp = \frac{e^2}{16 \pi^2} \intoq \ddx [ (2 - d) x \psl + d \mo ] \paren{ \frac{2}{\eps} - \gam + \ln(\frac{4\pi}{\Del}) }.
	}
	Then
	\al{
		\dv{\Sigw}{\psl} &= \frac{e^2}{16 \pi^2} \intoq \ddx \brac{ \frac{2}{\eps} - \gam + \ln(\frac{4\pi}{\Del}) } \dv{\psl}[ (2 - d) x \psl + d \mo ] \\
		&\hspace{20em} \phantom{=\ } + [ (2 - d) x \psl + d \mo ] \dv{\psl}( \frac{2}{\eps} - \ln(\Del) - \gam  + \ln(4\pi) ) \\[2ex]
		&= \frac{e^2}{16 \pi^2} \intoq \ddx \curly{ \brac{ \frac{2}{\eps} - \gam  + \ln(\frac{4\pi}{\Del}) } (2 - d) x + [ (2 - d) x \psl + d \mo ] \frac{2x (1 - x) \psl}{\Del} }
	}
	where
	\eq{
		\dv{\psl}( \ln\Del ) = \dv{\Del}{\psl} \dv{\Del}( \ln\Del )
		= -\frac{2x (1 - x) \psl}{\Del}
	}
	from Eqs.~\refeq{dDeldp} and \refeq{others}.  Then applying $m \approx \mo$ and discarding terms of $\order{\eps}$,
	\aln{
		\del\Zw &= \frac{e^2}{16 \pi^2} \intoq \ddx \curly{ (2 - d) x \brac{ \frac{2}{\eps} - \gam + \ln(\frac{4\pi}{\Delm}) } + \frac{2x (1 - x) [ (2 - d) x + d ] m^2}{\Delm} } \notag \\
		&= \frac{e^2}{8 \pi^2} \intoq \ddx \curly{ x \brac{ -\frac{2}{\eps} + \gam - \ln(\frac{4\pi}{\Delm}) } + \frac{2 m^2 x (x - 1) (x - 2)}{\Delm} }. \label{Zw1b}
	}
	
	To find $\del\Zq$, we need to start at Peskin \& Schroeder~(6.38) for arbitrary dimension:
	\aln{
		\ubpp \del\Gamm(p', p) \up &= \int\dddkf \frac{-i \gsnr}{(k - p)^2 + i \eps} \ubpp (-i e \gamn) \frac{i (\ksl' + m)}{{k'}^2 - m^2 + i \eps} \gamm \frac{i (\ksl + m)}{k^2 - m^2 + i \eps} (-i e \gamr) \up \notag \\
		&= -i e^2 \int\dddkf \ubpp \frac{\gamn (\ksl' + m) \gamm (\ksl + m) \gamsn}{[ (k - p)^2 + i \eps ] ({k'}^2 - m^2 + i \eps) (k^2 - m^2 + i \eps)} \up. \label{idek}
	}
	Using an expression analogous to Eq.~\refeq{FP}, we introduce the Feynman parameter $x$ and define $\ell = k - x p$.  We are implicitly using $q = k' - k = 0$.  We also define $\Del = x \mu^2 + (1 - x)^2 m^2$, where we have introduced a photon mass and let $m \approx \mo$.  The denominator of Eq.~\refeq{idek} is $D^3 = (\ell^2 + \Del + i \eps)^3$, and the numerator is
	\aln{
		N &= \ubpp \gamn (\lsl + x \psl + m) \gamm (\lsl + x \psl + m) \gamsn \up \notag \\
		&= \ubpp \gamn (\lsl \gamm \lsl + x \lsl \gamm \psl + m \lsl \gamm + x \psl \gamm \lsl + x^2 \psl \gamm \psl + m x \psl \gamm + m \gamm \lsl + m x \gamm \psl + m^2 \gamm) \gamsn \up \notag \\
		&\to \ubpp \gamn (\lsl \gamm \lsl + x^2 \psl \gamm \psl + 2 m x \gamm \psl + m^2 \gamm) \gamsn \up. \label{num}
	}
	To simplify each of these terms we apply~(7.87),
	\eqn{id1}{
		\int \dddlf \frac{\ellm \elln}{D^3} = \int \dddlf \frac{\gmn \ell^2}{d D^3},
	}
	(7.89),
	\eqn{id2}{
		\gamm \gamn \gamsm = -(2 - \eps) \gamn
		= -(d - 2) \gamn,
	}
	as well as~\cite[pp.~191--192]{Peskin}
	\al{ \label{id3}
		\psl \gamm &= 2 \pmu - \gamm \psl, &
		\psl \up &= \me \up, &
		\ubpp \psl' &= \ubpp \me,
	}
	and the Gordon identity
	\eqn{id4}{
		\ubpp \gamm \up = \ubpp \brac{ \frac{{\pmu}' + \pmu}{2 m} + \frac{i \sigmn \qsn}{2m} } \up
		\qimplies
		\pmu \to m \gamm.
	}
	Note that
	\al{
		\gamn \lsl \gamm \lsl \gamsn &= \gamn (2 \ellm - \gamm \lsl) \lsl \gamsn
		= \gamn (2 \ellm \ellr \gamsr - \gamm \lsl \lsl) \gamsn
		\to \gamn \paren{ \frac{2}{d} \ell^2 \gmr \gamsr - \gamm \ell^2 } \gamsn \\
		&\to \gamn \paren{ \frac{2 - d}{d} \ell^2 \gamm - 2 m^2 \gamm } \gamsn
		= \frac{(2 - d)^2}{d} \ell^2 \gamm + 2 (d - 2) m^2 \gamm, \\[2ex]
		%
		\gamn \psl \gamm \psl \gamsn &= \gamn (2 \pmu - \gamm \psl) (2 \psn - \gamsn \psl)
		= \gamn (4 \pmu \psn - 2 \pmu \gamsn \psl - 2 \gamm \psl \psn + \gamm \psl \gamsn \psl) \\
		&\to \gamn (4 m^2 \gamm \gamsn - 2 m^2 \gamm \gamsn - 2 m \gamm \psl \gamsn + m \gamm \psl \gamsn)
		= 2 m^2 \gamn \gamm \gamsn - m \gamn \gamm \psl \gamsn \\
		&= -2 m^2 (d - 2) \gamm - m \gamn \gamm (2 \psn - \gamsn \psl)
		\to -2 m^2 (d - 2) \gamm - m \gamn \gamm (2 m \gamsn - m \gamsn) \\
		&= -2 m^2 (d - 2) \gamm + m^2 (d - 2) \gamm
		= -m^2 (d - 2) \gamm, \\[2ex]
		%
		\gamn \gamm \psl \gamsn &= \gamn \gamm (2 \psn - \gamsn \psl)
		\to \gamn \gamm (2 m \gamsn - m \gamsn)
		= -m (d - 2) \gamm.
	}
